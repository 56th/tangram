284 inline const Vector<3> cross(const Vector<3>& a, const Vector<3>& b) {
285  Vector<3> r;
286  r[0] = a[1] * b[2] - a[2] * b[1];
287  r[1] = a[2] * b[0] - a[0] * b[2];
288  r[2] = a[0] * b[1] - a[1] * b[0];
289  return r;
290 }
