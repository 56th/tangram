154  xmof_ir->construct_minimesh(cellID);
155  const XMOF2D::MiniMesh& cminimesh = mesh.get_cell(cellID).get_minimesh();
156  for (int isc = 0; isc < cminimesh.ncells(); isc++) {

158  int nvrts = subcell.nfaces();

160  std::vector<Tangram::Entity_kind> vrts_parentkind(nvrts);

162  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
163  subcell_vrts[ivrt] = Tangram::Point<Dim>(subcell.get_node_crd(ivrt).x,
164  subcell.get_node_crd(ivrt).y);
165  vrts_iparent[ivrt] = subcell.get_node(ivrt).iparent();
166  vrts_parentkind[ivrt] = (vrts_iparent[ivrt] == -1) ?
167  Tangram::Entity_kind::UNKNOWN_KIND : Tangram::Entity_kind::NODE;
168  }

172  sides_iparent[iside] = subcell.get_face(iside).iparent();

176  }
177  (*cell_mat_poly).add_matpoly(subcell.get_material_index(), nvrts,
178  &subcell_vrts[0],

180  &sides_parentkind[0], &sides_iparent[0]);
181  }
182  }
183  else {

185  int nvrts = ccell.nfaces();

187  for (int ivrt = 0; ivrt < nvrts; ivrt++)

190  (*cell_mat_poly).add_matpoly(xmof_ir->get_cell_materials(cellID)[0],

193  }
194 
195  return std::shared_ptr<CellMatPoly<Dim>>(cell_mat_poly);
