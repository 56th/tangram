133  assert(num_matpolys_ > 0);
134  assert(matpoly_id < num_matpolys_);
135  std::vector<int> matverts;
136  std::vector<int> const& matfaces = matpoly_faces_[matpoly_id];
137  int nf = matfaces.size();
138  for (int jf = 0; jf < nf; jf++) {
139  int f = matfaces[jf];
140  std::vector<int> const& fverts = matface_vertices_[f];
141  if (D != 2) { // order does not matter
142  for (auto const& fv : fverts)
143  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())
144  matverts.push_back(fv);
145  } else { // order is important
146  if (matpoly_facedirs_[matpoly_id][jf] == 1) { // natural order
147  for (auto const& fv : fverts)

152  auto const &fv = *it;
153  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())

507 void CellMatPoly<D>::add_matpoly(int const matid,
508  double const matpoly_point0,
509  double const matpoly_point1,
510  Entity_kind const point0_parentkind,
511  Entity_kind const point1_parentkind,
512  int const point0_parentid,
513  int const point1_parentid) {

515  assert(D == 1);
516 
517  int new_matpoly_id = num_matpolys_;
518  if (std::find(cell_materialids_.begin(), cell_materialids_.end(), matid) ==
519  cell_materialids_.end())
520  cell_materialids_.push_back(matid);
521 
522  materialids_.push_back(matid);

524  matpoly_facedirs_.resize(num_matpolys_+1);

526  // Add matpoly points if they are not already there
527  // In 1D, faces are the same as vertices/points

529  bool found = false;

531  for (int i = 0; i < nv_ini; i++) {
532  Point<1>& p = matvertex_points_[i];

534  found = true; // This point is already in CellMatPoly

542  }

544  matvertex_points_.push_back(Point<1>(matpoly_point0));
545  matvertex_parentkind_.push_back(point0_parentkind);
546  matvertex_parentid_.push_back(point0_parentid);

550  matpoly_facedirs_[new_matpoly_id].push_back(0);

559  matface_parentid_.push_back(point0_parentid);

561  num_matverts_++;
562  num_matfaces_++;
563  }

569  found = true; // This point is already in CellMatPoly

571  matpoly_facedirs_[new_matpoly_id].push_back(1);
572

577  }

580  matvertex_parentkind_.push_back(point1_parentkind);
581  matvertex_parentid_.push_back(point1_parentid);

620 template<int D>
621 void CellMatPoly<D>::add_matpoly(int const matid,
622  int const numverts,
623  Point<2> const * const matpoly_points,
624  Entity_kind const * const points_parentkind,
625  int const * const points_parentid,
626  Entity_kind const * const faces_parentkind,
627  int const * const faces_parentid) {
628  assert(D == 2);
629 
630  int new_matpoly_id = num_matpolys_;
631  if (std::find(cell_materialids_.begin(), cell_materialids_.end(), matid) ==
632  cell_materialids_.end())
633  cell_materialids_.push_back(matid);
634 
635  materialids_.push_back(matid);
636 
637  std::vector<int> matverts(numverts);
638 
639  int nv_ini = num_matverts_;
640 
641  // Add matpoly points if they are not already there
642 
643  for (int j = 0; j < numverts; j++) {
644  Point<D> const& pmat = matpoly_points[j];
645 
646  bool found = false;
647  for (int i = 0; i < nv_ini; i++) { // Note, nv remains number of original pnts
648  Point<D>& p = matvertex_points_[i];
649  // We use exact equality criterium to identify coincident nodes.
650  // The use of distance tolerances can result in material poly's with
651  // degeneracies depending on the intersector employed (e.g. r2d) and is
652  // therefore not recommended.
653  if (p == pmat) {
654  found = true; // This point is already in CellMatPoly
655  matverts[j] = i;
656  break;
657  }
658  }
659 
660  if (!found) { // Introduce this point into CellMatPoly
661  matvertex_points_.push_back(pmat);
662  Entity_kind parentkind = points_parentkind ? points_parentkind[j] :
663  Entity_kind::UNKNOWN_KIND;
664  matvertex_parentkind_.push_back(parentkind);
665  int parentid = points_parentid ? points_parentid[j] : -1;
666  matvertex_parentid_.push_back(parentid);
667  matverts[j] = matvertex_points_.size()-1;
668  num_matverts_++;
669  }
670  }
671 
672  // Add faces
673  int numfaces = numverts; // In 2D, number of "faces", which are really edges,
674  // // is the same as number of vertices
675 
676  std::vector<int> matfaces(numfaces);
677  std::vector<int> matfacedirs(numfaces);
678 
679  int nmf_ini = num_matfaces_;
680  for (int j = 0; j < numfaces; j++) {
681  std::vector<int> mfverts(2);
682  mfverts[0] = matverts[j];
683  mfverts[1] = matverts[(j+1)%numverts];
684 
685  bool found = false;
686  for (int i = 0; i < nmf_ini; i++) {
687  std::vector<int> const& mfverts2 = matface_vertices_[i];
688 
689  if (mfverts[0] == mfverts2[1] && mfverts[1] == mfverts2[0]) {
690  found = true;
691  matfaces[j] = i;
692  matfacedirs[j] = 0;
693  assert(matface_matpolys_[i][1] == -1); // verify slot is empty
694  matface_matpolys_[i][1] = new_matpoly_id;
695  break;
696  }
697  }
698 
699  if (!found) { // create a face
700  matface_vertices_.push_back(mfverts);
701  Entity_kind parentkind = faces_parentkind ? faces_parentkind[j] :
702  Entity_kind::UNKNOWN_KIND;
703  matface_parentkind_.push_back(parentkind);
704  int parentid = faces_parentid ? faces_parentid[j] : -1;
705  matface_parentid_.push_back(parentid);
706  matfaces[j] = num_matfaces_;
707  matfacedirs[j] = 1;
708 
709  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
710  matface_matpolys_[num_matfaces_][0] = new_matpoly_id;
711  num_matfaces_++;
712  }
713  }
714 
715  matpoly_faces_.push_back(matfaces);
716  matpoly_facedirs_.push_back(matfacedirs);
717 
718 
719  // Geometric center of polygon; Assume that polygon is NOT so
720  // non-convex that the center is outside the polygon
721 
722  Point<D> cen;
723  for (int j = 0; j < numverts; j++)
724  cen += matpoly_points[j];
725  cen /= numverts;
726 
727  double volume = 0;
728  Point<D> centroid;
729  for (int j = 0; j < numverts; j++) {
730  Vector<D> vec0 = matpoly_points[(j+1)%numverts] - matpoly_points[j];
731  Vector<D> vec1 = cen - matpoly_points[j];
732  double trivolume = 0.5*cross(vec0, vec1);
733  volume += trivolume;
734  Point<D> tricen = (matpoly_points[j] + matpoly_points[(j+1)%numverts] +
735  cen)/3.0;
736  centroid += trivolume*tricen;
737  }
738  matpoly_volumes_.push_back(volume);
739 
740  centroid /= volume;
741  matpoly_centroids_.push_back(centroid);
742 
743  num_matpolys_++;
744 } // add_matpoly for 2D

809  Entity_kind parentkind = points_parentkind ? points_parentkind[j] :
810  Entity_kind::UNKNOWN_KIND;
811  matvertex_parentkind_.push_back(parentkind);
812  int parentid = points_parentid ? points_parentid[j] : -1;
813  matvertex_parentid_.push_back(parentid);
814  matverts[j] = num_matverts_;
815  num_matverts_++;
816  }
817  }
818 
819  // Add faces
820 
821  std::vector<int> matfaces(numfaces);
822  std::vector<int> matfacedirs(numfaces);
823 
824  int nmf_ini = num_matfaces_;
825 
826  int offset = 0;
827  for (int j = 0; j < numfaces; j++) {
828  int nmfv = numfaceverts[j];
829 
830  std::vector<int> mfverts(nmfv);
831  for (int i = 0; i < nmfv; i++)
832  mfverts[i] = matverts[faces_point_ids[offset+i]];
833 
834  offset += nmfv;
835 
836  // First check if an existing face matches the points of this face.
837  // Since we stipulated that the faces of the input material
838  // polyhedron to this routine must be specified such that their
839  // normal points out of the polyhedron, a face, if it exists, must
840  // point out of a previously created polyhedron and into the one
841  // that we are trying to create. Thus we check for a matching face
842  // with vertices reversed
843 
844  bool found = false;
845  for (int i = 0; i < nmf_ini; i++) {
846 
847  // Get vertices of existing face
848 
849  std::vector<int> const& mfverts2 = matface_vertices_[i];
850  int nmfv2 = mfverts2.size();
851  if (nmfv != nmfv2) continue; // No match if number of verts don't match
852 
853  // Check if all vertices of the two faces match
854 
855  for (int k = 0; k < nmfv; k++) {
856  if (mfverts[0] == mfverts2[k]) {
857 
858  // Found one matching vertex
859  // Check if the rest of the face vertices match (in reverse
860  // since the two material polyhedra will use the face in
861  // opposite directions)
862 
863  bool allmatch = true;
864  for (int m = 1; m < nmfv; m++) {
865  if (mfverts[m] != mfverts2[(k-m+nmfv)%nmfv]) {
866  allmatch = false;
867  break;
868  }
869  }
870 
871  // Found a matching face
872  if (allmatch) {
873  found = true;
874  matfaces[j] = i;
875  matfacedirs[j] = 0;
876  assert(matface_matpolys_[i][1] == -1);
877  matface_matpolys_[i][1] = new_matpoly_id;
878  break;
879  }
880  }
881  }
882 
883  if (found)
884  break;
885  } // for (int i = 0; i < ncmf_ini; i++)
886 
887  if (!found) { // create a new matface
888  matface_vertices_.push_back(mfverts);
889  Entity_kind parentkind = faces_parentkind ? faces_parentkind[j] :
890  Entity_kind::UNKNOWN_KIND;
891  matface_parentkind_.push_back(parentkind);
892  int parentid = faces_parentid ? faces_parentid[j] : -1;
893  matface_parentid_.push_back(parentid);
894  matfaces[j] = num_matfaces_;
895  matfacedirs[j] = 1;
896 
897  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
898  matface_matpolys_[num_matfaces_][0] = new_matpoly_id;
899  num_matfaces_++;
900  }
901  }
902 
903  matpoly_faces_.push_back(matfaces);
904  matpoly_facedirs_.push_back(matfacedirs);
905 
906 
907  // Compute geometric center of points. Assume that the cell is NOT so
908  // distorted that this point is outside the cell
909 
910  Point<3> cen;
911  for (int j = 0; j < numverts; j++)
912  cen += matpoly_points[j];
913  cen /= numverts;
914 
915  double volume = 0.0;
916  Point<D> centroid;
917  for (int j = 0; j < numfaces; j++) {
918  Point<D> fcen;
919  int f = matfaces[j];
920  int dir = matfacedirs[j];
921  int nmfv = matface_vertices_[f].size();
922  for (int k = 0; k < nmfv; k++) {
923  int k2 = matface_vertices_[f][k];
924  fcen += matvertex_points_[k2];
925  }
926  fcen /= nmfv;
927 
928  for (int k = 0; k < nmfv; k++) {
929  int v1 = matface_vertices_[f][k];
930  int v2 = (dir == 1) ? matface_vertices_[f][(k+1)%nmfv] :
931  matface_vertices_[f][(k-1+nmfv)%nmfv];
932 
933  Point<3> p1 = matvertex_points_[v1];
934  Point<3> p2 = matvertex_points_[v2];
935  Vector<3> vec0 = p2 - p1;
936  Vector<3> vec1 = fcen - p1;
937  Vector<3> vec2 = cen - p1;
938  Vector<3> inward_normal = -1.0*cross(vec0, vec1);
939  double tetvolume = dot(inward_normal, vec2)/6.0;
940  volume += tetvolume;
941  Point<3> tetcentroid = (p1 + p2 + fcen + cen)/4.0;
942  centroid += tetcentroid*tetvolume;
943  }
944  }
945 
946  matpoly_volumes_.push_back(volume);
947 
948  centroid /= volume;
949  matpoly_centroids_.push_back(centroid);
950 
951  num_matpolys_++;
952 } // add_matpoly for 3D
