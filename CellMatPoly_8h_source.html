26  exactly equal, the respective vertices will be treated as separate nodes

28  recognized as coincident and cell will not be recognized as adjacent.

132  assert(num_matpolys_ > 0);
133  assert(matpoly_id < num_matpolys_);
134  std::vector<int> matverts;
135  std::vector<int> const& matfaces = matpoly_faces_[matpoly_id];
136  int nf = matfaces.size();
137  for (int jf = 0; jf < nf; jf++) {
138  int f = matfaces[jf];
139  std::vector<int> const& fverts = matface_vertices_[f];
140  if (D != 2) { // order does not matter
141  for (auto const& fv : fverts)
142  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())
143  matverts.push_back(fv);
144  } else { // order is important
145  if (matpoly_facedirs_[matpoly_id][jf] == 1) { // natural order
146  for (auto const& fv : fverts)
147  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())
148  matverts.push_back(fv);
149  } else { // reverse the order
150  for (auto it = fverts.rbegin(); it != fverts.rend(); it++) {
151  auto const &fv = *it;
152  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())
153  matverts.push_back(fv);

155  }
156  }

447  assert(cmp_mat_id < material_moments_.size());
448  material_moments_[cmp_mat_id].assign(D + 1, 0.0);

505 template<int D>
506 void CellMatPoly<D>::add_matpoly(int const matid,
507  double const matpoly_point0,
508  double const matpoly_point1,
509  Entity_kind const point0_parentkind,
510  Entity_kind const point1_parentkind,
511  int const point0_parentid,
512  int const point1_parentid) {
513 
514  assert(D == 1);
515 
516  int new_matpoly_id = num_matpolys_;
517  if (std::find(cell_materialids_.begin(), cell_materialids_.end(), matid) ==
518  cell_materialids_.end())
519  cell_materialids_.push_back(matid);
520 
521  materialids_.push_back(matid);
522  matpoly_faces_.resize(num_matpolys_+1);
523  matpoly_facedirs_.resize(num_matpolys_+1);
524 
525  // Add matpoly points if they are not already there
526  // In 1D, faces are the same as vertices/points
527 
528  bool found = false;
529  int nv_ini = num_matverts_;
530  for (int i = 0; i < nv_ini; i++) {
531  Point<1>& p = matvertex_points_[i];
532  if (p[0] == matpoly_point0) {
533  found = true; // This point is already in CellMatPoly
534  matpoly_faces_[new_matpoly_id].push_back(i);
535  matpoly_facedirs_[new_matpoly_id].push_back(0);
536 
537  assert(matface_matpolys_[i][1] == -1); // verify slot is empty
538  matface_matpolys_[i][1] = new_matpoly_id; // new matpoly is to right of "face"
539  break;
540  }
541  }
542  if (!found) { // Introduce this point into CellMatPoly
543  matvertex_points_.push_back(Point<1>(matpoly_point0));
544  matvertex_parentkind_.push_back(point0_parentkind);
545  matvertex_parentid_.push_back(point0_parentid);
546 
547  int vert0 = num_matverts_, face0 = num_matverts_;
548  matpoly_faces_[new_matpoly_id].push_back(face0);
549  matpoly_facedirs_[new_matpoly_id].push_back(0);
550 
551  int new_matface_id = num_matfaces_;
552  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
553  matface_matpolys_[new_matface_id][1] = new_matpoly_id; // new matpoly is to right of "face"
554 
555  matface_vertices_.resize(num_matfaces_+1);
556  matface_vertices_[new_matface_id].push_back(vert0);
557  matface_parentkind_.push_back(point0_parentkind);
558  matface_parentid_.push_back(point0_parentid);
559 
560  num_matverts_++;
561  num_matfaces_++;
562  }
563 
564  found = false;
565  for (int i = 0; i < nv_ini; i++) {
566  Point<1>& p = matvertex_points_[i];
567  if (p[0] == matpoly_point1) {
568  found = true; // This point is already in CellMatPoly
569  matpoly_faces_[new_matpoly_id].push_back(i);
570  matpoly_facedirs_[new_matpoly_id].push_back(1);
571 
572  assert(matface_matpolys_[i][0] == -1); // verify slot is empty
573  matface_matpolys_[i][0] = new_matpoly_id; // new matpoly is to left of "face"
574  break;
575  }
576  }
577  if (!found) { // Introduce this point into CellMatPoly
578  matvertex_points_.push_back(Point<1>(matpoly_point1));
579  matvertex_parentkind_.push_back(point1_parentkind);
580  matvertex_parentid_.push_back(point1_parentid);
581 
582  int vert1 = num_matverts_, face1 = num_matverts_;
583  matpoly_faces_[new_matpoly_id].push_back(face1);
584  matpoly_facedirs_[new_matpoly_id].push_back(1);
585 
586  int new_matface_id = num_matfaces_;
587 
588  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
589  matface_matpolys_[new_matface_id][0] = new_matpoly_id; // new matpoly is to left of "face"
590 
591  matface_vertices_.resize(num_matfaces_+1);
592  matface_vertices_[new_matface_id].push_back(vert1);
593  matface_parentkind_.push_back(point1_parentkind);
594  matface_parentid_.push_back(point1_parentid);
595 
596  num_matverts_++;
597  num_matfaces_++;
598  }
599 
600  Point<D> vec = matpoly_point1 - matpoly_point0;
601  matpoly_volumes_.push_back(vec[0]);
602 
603  Point<D> cen = (matpoly_point0 + matpoly_point1)/2.0;
604  matpoly_centroids_.push_back(cen);
605 
606  num_matpolys_++;
607 } // add_matpoly for 1D

619 template<int D>
620 void CellMatPoly<D>::add_matpoly(int const matid,
621  int const numverts,
622  Point<2> const * const matpoly_points,
623  Entity_kind const * const points_parentkind,
624  int const * const points_parentid,
625  Entity_kind const * const faces_parentkind,
626  int const * const faces_parentid) {
627  assert(D == 2);
628 
629  int new_matpoly_id = num_matpolys_;
630  if (std::find(cell_materialids_.begin(), cell_materialids_.end(), matid) ==
631  cell_materialids_.end())
632  cell_materialids_.push_back(matid);
633 
634  materialids_.push_back(matid);
635 
636  std::vector<int> matverts(numverts);
637 
638  int nv_ini = num_matverts_;
639 
640  // Add matpoly points if they are not already there
641 
642  for (int j = 0; j < numverts; j++) {
643  Point<D> const& pmat = matpoly_points[j];
644 
645  bool found = false;
646  for (int i = 0; i < nv_ini; i++) { // Note, nv remains number of original pnts
647  Point<D>& p = matvertex_points_[i];
648  // We use exact equality criterium to identify coincident nodes.
649  // The use of distance tolerances can result in material poly's with
650  // degeneracies depending on the intersector employed (e.g. r2d) and is
651  // therefore not recommended.
652  if (p == pmat) {
653  found = true; // This point is already in CellMatPoly
654  matverts[j] = i;
655  break;
656  }
657  }
658 
659  if (!found) { // Introduce this point into CellMatPoly
660  matvertex_points_.push_back(pmat);
661  Entity_kind parentkind = points_parentkind ? points_parentkind[j] :
662  Entity_kind::UNKNOWN_KIND;
663  matvertex_parentkind_.push_back(parentkind);
664  int parentid = points_parentid ? points_parentid[j] : -1;
665  matvertex_parentid_.push_back(parentid);
666  matverts[j] = matvertex_points_.size()-1;
667  num_matverts_++;
668  }
669  }
670 
671  // Add faces
672  int numfaces = numverts; // In 2D, number of "faces", which are really edges,
673  // // is the same as number of vertices
674 
675  std::vector<int> matfaces(numfaces);
676  std::vector<int> matfacedirs(numfaces);
677 
678  int nmf_ini = num_matfaces_;
679  for (int j = 0; j < numfaces; j++) {
680  std::vector<int> mfverts(2);
681  mfverts[0] = matverts[j];
682  mfverts[1] = matverts[(j+1)%numverts];
683 
684  bool found = false;
685  for (int i = 0; i < nmf_ini; i++) {
686  std::vector<int> const& mfverts2 = matface_vertices_[i];
687 
688  if (mfverts[0] == mfverts2[1] && mfverts[1] == mfverts2[0]) {
689  found = true;
690  matfaces[j] = i;
691  matfacedirs[j] = 0;
692  assert(matface_matpolys_[i][1] == -1); // verify slot is empty
693  matface_matpolys_[i][1] = new_matpoly_id;
694  break;
695  }
696  }
697 
698  if (!found) { // create a face
699  matface_vertices_.push_back(mfverts);
700  Entity_kind parentkind = faces_parentkind ? faces_parentkind[j] :
701  Entity_kind::UNKNOWN_KIND;
702  matface_parentkind_.push_back(parentkind);
703  int parentid = faces_parentid ? faces_parentid[j] : -1;
704  matface_parentid_.push_back(parentid);
705  matfaces[j] = num_matfaces_;
706  matfacedirs[j] = 1;
707 
708  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
709  matface_matpolys_[num_matfaces_][0] = new_matpoly_id;
710  num_matfaces_++;
711  }
712  }
713 
714  matpoly_faces_.push_back(matfaces);
715  matpoly_facedirs_.push_back(matfacedirs);
716 
717 
718  // Geometric center of polygon; Assume that polygon is NOT so
719  // non-convex that the center is outside the polygon
720 
721  Point<D> cen;
722  for (int j = 0; j < numverts; j++)
723  cen += matpoly_points[j];
724  cen /= numverts;
725 
726  double volume = 0;
727  Point<D> centroid;
728  for (int j = 0; j < numverts; j++) {
729  Vector<D> vec0 = matpoly_points[(j+1)%numverts] - matpoly_points[j];
730  Vector<D> vec1 = cen - matpoly_points[j];
731  double trivolume = 0.5*cross(vec0, vec1);
732  volume += trivolume;
733  Point<D> tricen = (matpoly_points[j] + matpoly_points[(j+1)%numverts] +
734  cen)/3.0;
735  centroid += trivolume*tricen;
736  }
737  matpoly_volumes_.push_back(volume);
738 
739  centroid /= volume;
740  matpoly_centroids_.push_back(centroid);
741 
742  num_matpolys_++;
743 } // add_matpoly for 2D

811  int parentid = points_parentid ? points_parentid[j] : -1;
812  matvertex_parentid_.push_back(parentid);
813  matverts[j] = num_matverts_;
814  num_matverts_++;
815  }
816  }
817 
818  // Add faces
819 
820  std::vector<int> matfaces(numfaces);
821  std::vector<int> matfacedirs(numfaces);
822 
823  int nmf_ini = num_matfaces_;
824 
825  int offset = 0;
826  for (int j = 0; j < numfaces; j++) {
827  int nmfv = numfaceverts[j];
828 
829  std::vector<int> mfverts(nmfv);
830  for (int i = 0; i < nmfv; i++)
831  mfverts[i] = matverts[faces_point_ids[offset+i]];
832 
833  offset += nmfv;
834 
835  // First check if an existing face matches the points of this face.
836  // Since we stipulated that the faces of the input material
837  // polyhedron to this routine must be specified such that their
838  // normal points out of the polyhedron, a face, if it exists, must
839  // point out of a previously created polyhedron and into the one
840  // that we are trying to create. Thus we check for a matching face
841  // with vertices reversed
842 
843  bool found = false;
844  for (int i = 0; i < nmf_ini; i++) {
845 
846  // Get vertices of existing face
847 
848  std::vector<int> const& mfverts2 = matface_vertices_[i];
849  int nmfv2 = mfverts2.size();
850  if (nmfv != nmfv2) continue; // No match if number of verts don't match
851 
852  // Check if all vertices of the two faces match
853 
854  for (int k = 0; k < nmfv; k++) {
855  if (mfverts[0] == mfverts2[k]) {
856 
857  // Found one matching vertex
858  // Check if the rest of the face vertices match (in reverse
859  // since the two material polyhedra will use the face in
860  // opposite directions)
861 
862  bool allmatch = true;
863  for (int m = 1; m < nmfv; m++) {
864  if (mfverts[m] != mfverts2[(k-m+nmfv)%nmfv]) {
865  allmatch = false;
866  break;
867  }
868  }
869 
870  // Found a matching face
871  if (allmatch) {
872  found = true;
873  matfaces[j] = i;
874  matfacedirs[j] = 0;
875  assert(matface_matpolys_[i][1] == -1);
876  matface_matpolys_[i][1] = new_matpoly_id;
877  break;
878  }
879  }
880  }
881 
882  if (found)
883  break;
884  } // for (int i = 0; i < ncmf_ini; i++)
885 
886  if (!found) { // create a new matface
887  matface_vertices_.push_back(mfverts);
888  Entity_kind parentkind = faces_parentkind ? faces_parentkind[j] :
889  Entity_kind::UNKNOWN_KIND;
890  matface_parentkind_.push_back(parentkind);
891  int parentid = faces_parentid ? faces_parentid[j] : -1;
892  matface_parentid_.push_back(parentid);
893  matfaces[j] = num_matfaces_;
894  matfacedirs[j] = 1;
895 
896  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
897  matface_matpolys_[num_matfaces_][0] = new_matpoly_id;
898  num_matfaces_++;
899  }
900  }
901 
902  matpoly_faces_.push_back(matfaces);
903  matpoly_facedirs_.push_back(matfacedirs);
904 
905 
906  // Compute geometric center of points. Assume that the cell is NOT so
907  // distorted that this point is outside the cell
908 
909  Point<3> cen;
910  for (int j = 0; j < numverts; j++)
911  cen += matpoly_points[j];
912  cen /= numverts;
913 
914  double volume = 0.0;
915  Point<D> centroid;
916  for (int j = 0; j < numfaces; j++) {
917  Point<D> fcen;
918  int f = matfaces[j];
919  int dir = matfacedirs[j];
920  int nmfv = matface_vertices_[f].size();
921  for (int k = 0; k < nmfv; k++) {
922  int k2 = matface_vertices_[f][k];
923  fcen += matvertex_points_[k2];
924  }
925  fcen /= nmfv;
926 
927  for (int k = 0; k < nmfv; k++) {
928  int v1 = matface_vertices_[f][k];
929  int v2 = (dir == 1) ? matface_vertices_[f][(k+1)%nmfv] :
930  matface_vertices_[f][(k-1+nmfv)%nmfv];
931 
932  Point<3> p1 = matvertex_points_[v1];
933  Point<3> p2 = matvertex_points_[v2];
934  Vector<3> vec0 = p2 - p1;
935  Vector<3> vec1 = fcen - p1;
936  Vector<3> vec2 = cen - p1;
937  Vector<3> inward_normal = -1.0*cross(vec0, vec1);
938  double tetvolume = dot(inward_normal, vec2)/6.0;
939  volume += tetvolume;
940  Point<3> tetcentroid = (p1 + p2 + fcen + cen)/4.0;
941  centroid += tetcentroid*tetvolume;
942  }
943  }
944 
945  matpoly_volumes_.push_back(volume);
946 
947  centroid /= volume;
948  matpoly_centroids_.push_back(centroid);
949 
950  num_matpolys_++;
951 } // add_matpoly for 3D
