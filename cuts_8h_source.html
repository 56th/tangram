49 void r3d_poly_components(const r3d_poly& r3dpoly, std::vector<r3d_poly>& poly_components) {
50  poly_components.clear();
51 
52  int total_nvrts = static_cast<int>(r3dpoly.nverts);
53  std::vector<int> vrt_marks(total_nvrts, 0);
54  std::vector< std::vector<bool> > walked_edge(total_nvrts);
55  for (int ivrt = 0; ivrt < total_nvrts; ivrt++)
56  walked_edge[ivrt].assign(3, false);
57 
58  std::vector<bool> vrt_in_cmp(total_nvrts, false);
59  std::vector<int> poly_irvt2cmp_ivrt(total_nvrts, -1);
60 
61  int icmp_first_vrt = 0;
62  r3d_real cmp_moments[R3D_NUM_MOMENTS(R3D_POLY_ORDER)];
63 
64  while(icmp_first_vrt < total_nvrts) {
65  r3d_poly cur_cmp;
66  cur_cmp.nverts = 0;
67 
68  //Walk all faces of the current components
69  int iface_first_vrt = icmp_first_vrt;
70  int iedge = 0;
71  while (iface_first_vrt < total_nvrts) {
72  int icur_vrt = iface_first_vrt;
73 
74  for (iedge = 0; iedge < 3; iedge++)
75  if (!walked_edge[iface_first_vrt][iedge])
76  break;
77 
78  //Walk the current face
79  do {
80  vrt_marks[icur_vrt]++;
81 
82  if (!vrt_in_cmp[icur_vrt]) {
83  poly_irvt2cmp_ivrt[icur_vrt] = cur_cmp.nverts;
84  cur_cmp.nverts++;
85  cur_cmp.verts[poly_irvt2cmp_ivrt[icur_vrt]] = r3dpoly.verts[icur_vrt];
86  vrt_in_cmp[icur_vrt] = true;
87  }
88 
89  int inext_vrt = r3dpoly.verts[icur_vrt].pnbrs[iedge];
90  walked_edge[icur_vrt][iedge] = true;
91 
92  int ireturn_edge;
93  for (ireturn_edge = 0; ireturn_edge < 3; ireturn_edge++)
94  if (r3dpoly.verts[inext_vrt].pnbrs[ireturn_edge] == icur_vrt)
95  break;
96 
97  iedge = (ireturn_edge + 2)%3;
98  icur_vrt = inext_vrt;
99 
100  } while (icur_vrt != iface_first_vrt);
101 
102  for (iface_first_vrt = 0; iface_first_vrt < total_nvrts; iface_first_vrt++)
103  if( (vrt_marks[iface_first_vrt] > 0) && (vrt_marks[iface_first_vrt] < 3) )
104  break;
105  }
106 
107  //Neighbors data still uses r3dpoly indices
108  for (int icmp_vrt = 0; icmp_vrt < cur_cmp.nverts; icmp_vrt++)
109  for (int iedge = 0; iedge < 3; iedge++) {
110  int old_id = cur_cmp.verts[icmp_vrt].pnbrs[iedge];
111  cur_cmp.verts[icmp_vrt].pnbrs[iedge] = poly_irvt2cmp_ivrt[old_id];
112  }
113 
114  //Confirm that the component is not empty
115  r3d_reduce(&cur_cmp, cmp_moments, R3D_POLY_ORDER);
116  if (cmp_moments[0] > std::numeric_limits<double>::epsilon())
117  poly_components.push_back(cur_cmp);
118 
119  //Find the first vertex of the next component
120  for (icmp_first_vrt = 0; icmp_first_vrt < total_nvrts; icmp_first_vrt++)
121  if (vrt_marks[icmp_first_vrt] == 0)
122  break;
123  }
124 }

166  R3DPOLY::Position operator()(const int checkID) const {
167  int polyID = IDs_to_check_[checkID];
168  const r3d_poly& r3dpoly = polys_data_[polyID]->r3dpoly;
169 
170  bool intersects = false;
171  for (int ivrt = 0; ivrt < r3dpoly.nverts; ivrt++) {
172  Tangram::Point3 cur_vrt;
173  for (int ixyz = 0; ixyz < 3; ixyz++)
174  cur_vrt[ixyz] = r3dpoly.verts[ivrt].pos.xyz[ixyz];
175 
176  if (Tangram::point_inside_matpoly(convex_matpoly_, cur_vrt, true)) {
177  if (!intersects && (ivrt > 0)) return R3DPOLY::Position::INTERSECTS;
178  intersects = true;
179  }
180  else if (intersects) return R3DPOLY::Position::INTERSECTS;
181  }
182 
183  if (intersects) return R3DPOLY::Position::INSIDE;
184 
185  //Instead of checking if MatPoly vertices are inside the r3d_poly,
186  //we look at the volume of the actual intersection
187  r3d_poly intersection = r3dpoly;
188  for (int iplane = 0; iplane < face_planes_.size(); iplane++) {
189  r3d_clip(&intersection, &face_planes_[iplane], 1);
190  if (intersection.nverts == 0) return R3DPOLY::Position::OUTSIDE;
191  }
192 
193  return R3DPOLY::Position::INTERSECTS;
194  }

232  PolyHalfspaceData_t operator()(const int polyID) const {
233  PolyHalfspaceData_t hs_data = {.lower = std::make_shared<RefPolyData_t>(),
234  .upper = std::make_shared<RefPolyData_t>() };
235 
236  const r3d_poly& r3dpoly = polys_data_[polyID]->r3dpoly;
237  const r3d_real* poly_moments = polys_data_[polyID]->moments;
238  int cellID = polys_data_[polyID]->cellID;
239 
240  r3d_poly split_poly = r3dpoly;
241  r3d_split(&split_poly, 1, cutting_plane_, &hs_data.upper->r3dpoly, &hs_data.lower->r3dpoly);
242 
243 #if BUGS_IN_R3D
244  std::shared_ptr<RefPolyData_t> hs_data_ptrs[2] = {hs_data.lower, hs_data.upper};
245  for (int ihs = 0; ihs < 2; ihs++) {
246  if (hs_data_ptrs[ihs]->r3dpoly.nverts > 0) {
247  std::vector<r3d_poly> poly_components;
248  r3d_poly_components(hs_data_ptrs[ihs]->r3dpoly, poly_components);
249  if (poly_components.empty())
250  hs_data_ptrs[ihs]->r3dpoly.nverts = 0;
251  else {
252  assert(poly_components.size() == 1);
253  if (hs_data_ptrs[ihs]->r3dpoly.nverts != poly_components[0].nverts)
254  hs_data_ptrs[ihs]->r3dpoly = poly_components[0];
255  hs_data_ptrs[ihs]->cellID = cellID;
256  r3d_reduce(&hs_data_ptrs[ihs]->r3dpoly, hs_data_ptrs[ihs]->moments, R3D_POLY_ORDER);
257  }
258  }
259  }
260 #else
261  bool nnz_cutoff = false;
262  if (hs_data.lower->r3dpoly.nverts > 0) {
263  r3d_reduce(&hs_data.lower->r3dpoly, hs_data.lower->moments, R3D_POLY_ORDER);
264  if (hs_data.lower->moments[0] > std::numeric_limits<double>::epsilon()) {
265  hs_data.lower->cellID = cellID;
266  nnz_cutoff = true;
267  }
268  else
269  hs_data.lower->r3dpoly.nverts = 0;
270  }
271 
272  if (hs_data.upper->r3dpoly.nverts > 0) {
273  if (nnz_cutoff) {
274  for (int im = 0; im < R3D_NUM_MOMENTS(R3D_POLY_ORDER); im++)
275  hs_data.upper->moments[im] =
276  poly_moments[im] - hs_data.lower->moments[im];
277  }
278  else {
279  for (int im = 0; im < R3D_NUM_MOMENTS(R3D_POLY_ORDER); im++)
280  hs_data.upper->moments[im] = poly_moments[im];
281  }
282  if (hs_data.upper->moments[0] > std::numeric_limits<double>::epsilon())
283  hs_data.upper->cellID = cellID;
284  else
285  hs_data.upper->r3dpoly.nverts = 0;
286  }
287 #endif
288  return hs_data;
289  }

394  const Tangram::MatPoly<3>& convex_matpoly,

421 
422  Tangram::transform(Tangram::make_counting_iterator(0),
423  Tangram::make_counting_iterator(num_in_box),
424  check_result.begin(), r3d_isect_check);
425 
426  iinterior_polys.clear();
427  iinstersecting_polys.clear();
428  for (int iip = 0; iip < num_in_box; iip++)
429  switch (check_result[iip]) {
430  case R3DPOLY::Position::OUTSIDE:
431  iexterior_polys.push_back(iin_box_polys[iip]);
432  break;
433  case R3DPOLY::Position::INTERSECTS:
434  iinstersecting_polys.push_back(iin_box_polys[iip]);
435  break;
436  case R3DPOLY::Position::INSIDE:
437  iinterior_polys.push_back(iin_box_polys[iip]);
438  break;
439  default: throw std::runtime_error("Undefined position");
440  }

548  new_exterior_data;
549  //Face normal is pointing outward

551  upd_intersection_data, new_exterior_data);
552 
553  intersection_data = upd_intersection_data;
554  exterior_data.reserve(exterior_data.size() + new_exterior_data.size());
555  exterior_data.insert(exterior_data.end(),
556  new_exterior_data.begin(), new_exterior_data.end());
