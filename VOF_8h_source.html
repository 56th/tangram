111  void get_plane_position(const int cellID,
112  const int matID,
113  const std::vector< MatPoly<Dim> >& mixed_polys,
114  Plane_t<Dim>& cutting_plane,
115  const bool planar_faces = true) const {
116  double vol_tol = im_tols_.fun_eps;
117 
118  std::vector<int> istencil_cells;
119  mesh_.cell_get_node_adj_cells(cellID, Entity_type::ALL, &istencil_cells);
120  istencil_cells.insert(istencil_cells.begin(), cellID);
121  int nsc = (int) istencil_cells.size();
122 
123  // Create stencil for the volume fractions gradient: the first entry corresponds
124  // to the current cell, the rest correspond to all its neighbors through the nodes
125  std::vector<double> stencil_vfracs(nsc);
126  std::vector< Point<Dim> > stencil_centroids(nsc);
127  for (int isc = 0; isc < nsc; isc++) {
128  const std::vector<int>& cur_mat_ids = cell_mat_ids_[istencil_cells[isc]];
129  int local_id = (int) (std::find(cur_mat_ids.begin(), cur_mat_ids.end(), matID) -
130  cur_mat_ids.begin());
131  if (local_id != cur_mat_ids.size())
132  stencil_vfracs[isc] = cell_mat_vfracs_[istencil_cells[isc]][local_id];
133 
134  mesh_.cell_centroid(istencil_cells[isc], &stencil_centroids[isc]);
135  }
136 
137  double target_vol = stencil_vfracs[0]*mesh_.cell_volume(cellID);
138 
139  // Use least squares to compute the gradient
140  cutting_plane.normal = -ls_gradient(stencil_centroids, stencil_vfracs);
141  double grad_norm = cutting_plane.normal.norm();
142  if (is_equal(grad_norm, 0.0)) {
143  // Zero gradient: we choose SLIC-like plane orientation
144  // Normal is set in the direction of the x-axis
145  cutting_plane.normal.axis(0);
146  }
147  else
148  cutting_plane.normal /= grad_norm;
149 
150  //Create cutting distance solver
151  CuttingDistanceSolver<Dim, MatPoly_Clipper>
152  solve_cut_dst(mixed_polys, cutting_plane.normal, im_tols_, planar_faces);
153 
154  solve_cut_dst.set_target_volume(target_vol);
155  std::vector<double> clip_res = solve_cut_dst();
156  cutting_plane.dist2origin = clip_res[0];
157 
158 #ifdef DEBUG
159  // Check if the resulting volume matches the reference value
160  double cur_vol_err = std::fabs(clip_res[1] - target_vol);
161  if (cur_vol_err > vol_tol)
162  std::cerr << "VOF for cell " << cellID << ": after " << im_tols_.max_num_iter <<
163  " iteration(s) achieved error in volume for material " <<
164  matID << " is " << cur_vol_err << ", volume tolerance is " << vol_tol << std::endl;
165 #endif
166  }

173  std::shared_ptr<CellMatPoly<Dim>> operator()(const int cell_op_ID) const {
174  int cellID = icells_to_reconstruct[cell_op_ID];
175 
176  // Check if the cell is single-material
177  if (cell_mat_ids_[cellID].size() == 1) {
178  std::shared_ptr< CellMatPoly<Dim> > cmp_ptr(new CellMatPoly<Dim>(cellID));
179  MatPoly<Dim> cell_matpoly;
180  cell_get_matpoly(mesh_, cellID, &cell_matpoly);
181  cell_matpoly.set_mat_id(cell_mat_ids_[cellID][0]);
182  cmp_ptr->add_matpoly(cell_matpoly);
183 
184  return cmp_ptr;
185  }
186 
187  // Use the nested dissections algorithm for multi-material cells.
188  // Note that nested dissections uses this instance of reconstructor
189  // to invoke get_plane_position position method. Nested dissections
190  // itself does not have its own MeshWrapper, MatPoly_Clipper, etc.,
191  // they all are reconstructor specific
192  NestedDissections<VOF, Dim, MatPoly_Splitter>
193  nested_dissections(*this, cellID, all_convex_);
194 
195  // We clip material in the same order they are given for the cell
196  int nmats = (int) cell_mat_ids_[cellID].size();
197  std::vector<int> direct_order(nmats);
198  std::iota(direct_order.begin(), direct_order.end(), 0);
199  // Note that this is the order of local materials, not material
200  // indices. Nested dissections uses cell_materials method to get
201  // actual material indices.
202  nested_dissections.set_cell_materials_order(direct_order);
203 
204  return nested_dissections();
205  }
