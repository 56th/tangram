122  void get_plane_position(const int cellID,
123  const int matID,
124  const std::vector< MatPoly<Dim> >& mixed_polys,
125  Plane_t<Dim>& cutting_plane,
126  const bool planar_faces = true) const {
127  double vol_tol = ims_tols_[0].fun_eps;
128 
129  std::vector<int> istencil_cells;
130  mesh_.cell_get_node_adj_cells(cellID, Entity_type::ALL, &istencil_cells);
131  istencil_cells.insert(istencil_cells.begin(), cellID);
132  int nsc = static_cast<int>(istencil_cells.size());
133 
134  // Create stencil for the volume fractions gradient: the first entry corresponds
135  // to the current cell, the rest correspond to all its neighbors through the nodes
136  std::vector<double> stencil_vfracs(nsc);
137  std::vector< Point<Dim> > stencil_centroids(nsc);
138  for (int isc = 0; isc < nsc; isc++) {
139  const std::vector<int>& cur_mat_ids = cell_mat_ids_[istencil_cells[isc]];
140  int local_id = std::distance(cur_mat_ids.begin(),
141  std::find(cur_mat_ids.begin(), cur_mat_ids.end(), matID));
142 
143  if (local_id != cur_mat_ids.size())
144  stencil_vfracs[isc] = cell_mat_vfracs_[istencil_cells[isc]][local_id];
145 
146  mesh_.cell_centroid(istencil_cells[isc], &stencil_centroids[isc]);
147  }
148 
149  double target_vol = stencil_vfracs[0]*mesh_.cell_volume(cellID);
150 
151  // Use least squares to compute the gradient
152  cutting_plane.normal = -ls_gradient(stencil_centroids, stencil_vfracs);
153  double grad_norm = cutting_plane.normal.norm();
154  if (is_equal(grad_norm, 0.0)) {
155  // Zero gradient: we choose SLIC-like plane orientation
156  // Normal is set in the direction of the x-axis
157  cutting_plane.normal.axis(0);
158  }
159  else
160  cutting_plane.normal /= grad_norm;
161 
162  //Create cutting distance solver
163  CuttingDistanceSolver<Dim, MatPoly_Clipper>
164  solve_cut_dst(mixed_polys, cutting_plane.normal, ims_tols_[0], planar_faces);
165 
166  solve_cut_dst.set_target_volume(target_vol);
167  std::vector<double> clip_res = solve_cut_dst();
168  cutting_plane.dist2origin = clip_res[0];
169 
170 #ifdef DEBUG
171  // Check if the resulting volume matches the reference value
172  double cur_vol_err = std::fabs(clip_res[1] - target_vol);
173  if (cur_vol_err > vol_tol)
174  std::cerr << "VOF for cell " << cellID << ": after " << ims_tols_[0].max_num_iter <<
175  " iteration(s) achieved error in volume for material " <<
176  matID << " is " << cur_vol_err << ", volume tolerance is " << vol_tol << std::endl;
177 #endif
178  }

188  // Check if the cell is single-material

191  MatPoly<Dim> cell_matpoly;

199  // Use the nested dissections algorithm for multi-material cells.
200  // Note that nested dissections uses this instance of reconstructor
201  // to invoke get_plane_position position method. Nested dissections
202  // itself does not have its own MeshWrapper, MatPoly_Clipper, etc.,
203  // they all are reconstructor specific
204  NestedDissections<VOF, Dim, MatPoly_Splitter>
205  nested_dissections(*this, cellID, all_convex_);
206
