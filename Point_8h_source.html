233 template <long D> inline bool
234 approxEq(const Point<D>& p1, const Point<D>& p2, double tol = 1.0e-8) {
235  // This uses the L_infty norm to avoid nearby subtractions
236  for (int i = 0; i < D; i++)
237  if (p1[i] < (p2[i] - tol) || p1[i] > (p2[i] + tol))
238  return false;
239  return true;
240 }

242 template <long D> bool
243 operator<(const Point<D>& p1, const Point<D>& p2) {
244  if (approxEq(p1, p2))
245  return false;
246  else
247  for (int i = 0; i < D; ++i) {
248  if (p1[i] < p2[i])

250  else if (p2[i] < p1[i])
251  return false;
252  }
253  return false;
