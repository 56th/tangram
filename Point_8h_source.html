239 template <long D> inline bool
240 approxEq(const Point<D>& p1, const Point<D>& p2, double tol = 1.0e-8) {
241  // This uses the L_infty norm to avoid nearby subtractions
242  for (int i = 0; i < D; i++)
243  if (p1[i] < (p2[i] - tol) || p1[i] > (p2[i] + tol))
244  return false;
245  return true;
246 }
247 
248 template <long D> bool
249 operator<(const Point<D>& p1, const Point<D>& p2) {
250  if (approxEq(p1, p2))
251  return false;
252  else
253  for (int i = 0; i < D; ++i) {
254  if (p1[i] < p2[i])
255  return true;
256  else if (p2[i] < p1[i])
257  return false;
258  }
259  return false;
260 }
