98  std::shared_ptr<CellMatPoly<Dim>> operator()(const int cell_op_ID) const {
99  double vol_tol = im_tols_.fun_eps; // Volume tolerance
100 
101  int cellID = icells_to_reconstruct[cell_op_ID];
102  auto numMats = cell_num_mats_[cellID];
103 
104  CellMatPoly<Dim>* cellpoly = new CellMatPoly<Dim>(cellID);
105 
106  auto iStart = cell_mat_offsets_[cellID];
107 
108  //Sets of MatPoly's on two sides of the cutting plane
109  HalfSpaceSets_t<Dim> hs_sets;
110 
111  // For every chopped off material, we split MatPoly's above the plane
112  hs_sets.upper_halfspace_set.matpolys.resize(1);
113  Tangram::cell_get_matpoly(mesh_, cellID, &hs_sets.upper_halfspace_set.matpolys[0]);
114 
115  // Just going along x-direction
116  Plane_t<Dim> cutting_plane;
117  cutting_plane.normal = Vector3(1.0, 0.0, 0.0);
118 
119  //Create Splitter instance
120  MatPoly_Splitter split_matpolys(hs_sets.upper_halfspace_set.matpolys,
121  cutting_plane, all_convex_);
122 
123  //Create cutting distance solver: if not all cells are convex, we assume that
124  //faces are non-planar
125  CuttingDistanceSolver<Dim, Tangram::ClipR3D>
126  solve_cut_dst(hs_sets.upper_halfspace_set.matpolys,
127  cutting_plane.normal, im_tols_, all_convex_);
128 
129  //Cutting from left to right
130  double cell_volume = mesh_.cell_volume(cellID);
131  for (int iMat(0); iMat < numMats; iMat++) {
132  double target_vol = cell_mat_volfracs_[iStart + iMat]*cell_volume;
133  // If the target volume is too small, skip it
134  if (target_vol < vol_tol) continue;
135 
136  const MatPolySet_t<Dim>* single_mat_set_ptr;
137  //On the last iteration the remaining part is single-material,
138  //so we don't need to split it
139  if (iMat == numMats - 1)
140  single_mat_set_ptr = &hs_sets.upper_halfspace_set;
141  else {
142  // Find distance to origin for the cutting plane
143  solve_cut_dst.set_target_volume(target_vol);
144  std::vector<double> clip_res = solve_cut_dst();
145  cutting_plane.dist2origin = clip_res[0];
146 
147 #ifdef DEBUG
148  // Check if the resulting volume matches the reference value
149  double cur_vol_err = std::fabs(clip_res[1] - target_vol);
150  if (cur_vol_err > vol_tol)
151  std::cerr << "SLIC for cell " << cellID << ": after " << im_tols_.max_num_iter <<
152  " iteration(s) achieved error in volume for material " <<
153  cell_mat_ids_[iStart + iMat] << " is " << cur_vol_err <<
154  ", volume tolerance is " << vol_tol << std::endl;
155 #endif
156  hs_sets = split_matpolys();
157 
158  //Chopped off single-material MatPoly's are below the plane
159  single_mat_set_ptr = &hs_sets.lower_halfspace_set;
160  }
161  //Add single-material MatPoly's to CellMatPoly
162  for (int ismp = 0; ismp < single_mat_set_ptr->matpolys.size(); ismp++) {
163  const MatPoly<Dim>& cur_matpoly = single_mat_set_ptr->matpolys[ismp];
164  // Flatten the face vertices for the single-material MatPoly
165  int nfaces = cur_matpoly.num_faces();
166  std::vector<int> nface_vrts(nfaces);
167  std::vector<int> faces_vrts;
168  for (int iface = 0; iface < nfaces; iface++) {
169  const std::vector<int>& face_ivrts = cur_matpoly.face_vertices(iface);
170  nface_vrts[iface] = face_ivrts.size();
171  faces_vrts.insert(faces_vrts.end(), face_ivrts.begin(), face_ivrts.end());
172  }
173  //Add the MatPoly below the cutting plane to CellMatPoly
174  (*cellpoly).add_matpoly(cell_mat_ids_[iStart + iMat],
175  cur_matpoly.num_vertices(),
176  &cur_matpoly.points()[0],
177  nullptr, nullptr,
178  nfaces, &nface_vrts[0], &faces_vrts[0],
179  nullptr, nullptr);
180  }
181  }
182 
183  return std::shared_ptr<CellMatPoly<Dim>>(cellpoly);
184  }

split_r3d.h
Routines for splitting convex or non-convex MatPoly&#39;s by a cutting plane. Uses R3D. A convex MatPoly will be split into two MatPoly&#39;s. A non-convex MatPoly will be decomposed into convex polyhedra using its centroid, several MatPoly&#39;s could be returned for each half-plane. All the faces of the input MatPoly are assumed to be planar and no additional facetization is performed, so when sequentially splitting of a faceted MatPoly, no additional facets will be introduced.
