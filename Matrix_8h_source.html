113  Matrix inverse() const {
114  if (Rows_ != Columns_) {
115  std::cerr << "Matrix is rectangular" << std::endl;
116  throw std::exception();
117  }

119  Matrix Ainv(Rows_, Rows_, 0.0);
120 
121  // Create a temporary matrix with twice as many columns
122  Matrix D(Rows_, 2*Rows_);
123 
124  // Initialize the reduction matrix
125  int Rows2 = 2*Rows_;
126  for (int i = 0; i < Rows_; i++) {
127  for (int j = 0; j < Rows_; j++) {
128  D[i][j] = A_[i*Columns_+j];
129  D[i][Rows_+j] = 0.0;
130  }
131  D[i][Rows_+i] = 1.0;
132  }
133 
134  // Do the reduction
135  for (int i = 0; i < Rows_; i++) {
136  double alpha = D[i][i];
137  if (alpha == 0.0) {
138  std::cerr << "invert: Singular Matrix" << std::endl;
139  return Ainv;
140  }
141 
142  for (int j = 0; j < Rows2; j++)
143  D[i][j] = D[i][j]/alpha;
144 
145  for (int k = 0; k < Rows_; k++) {
146  if ((k - i) == 0)
147  continue;
148 
149  double beta = D[k][i];
150  for (int j = 0; j < Rows2; j++)
151  D[k][j] = D[k][j] - beta*D[i][j];
152  }
153  }
154 
155  // Copy result into output matrix
156  for (int i = 0; i < Rows_; i++)
157  for (int j = 0; j < Rows_; j++)
158  Ainv[i][j] = D[i][j + Rows_];
159 
160  return Ainv;
161  }
