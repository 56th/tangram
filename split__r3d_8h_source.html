39 void
40 matpoly_to_r3dpoly(const MatPoly<3>& mat_poly,
41  r3d_poly& r3dpoly) {
42  //Translate coordinates of vertices to R3D format
43  const std::vector<Point3>& matpoly_vrts = mat_poly.points();
44  int nvrts = mat_poly.num_vertices();
45  r3d_rvec3* r3dized_poly_vrts;
46  r3dized_poly_vrts = new r3d_rvec3 [nvrts];
47  for (int ivrt = 0; ivrt < nvrts; ivrt++)
48  for (int ixyz = 0; ixyz < 3; ixyz++)
49  r3dized_poly_vrts[ivrt].xyz[ixyz] = matpoly_vrts[ivrt][ixyz];
50 
51  //Get faces of the polyhedron in terms of local indices of its vertices
52  int nfaces = mat_poly.num_faces();
53  r3d_int* nface_vrts;
54  nface_vrts = new r3d_int [nfaces];
55  r3d_int** r3dized_poly_faces;
56  r3dized_poly_faces = new r3d_int* [nfaces];
57  for (int iface = 0; iface < nfaces; iface++) {
58  const std::vector<int>& matpoly_faces = mat_poly.face_vertices(iface);
59  int face_nvrts = (int) matpoly_faces.size();
60  nface_vrts[iface] = face_nvrts;
61  r3dized_poly_faces[iface] = new r3d_int [face_nvrts];
62  for (int ivrt = 0; ivrt < face_nvrts; ivrt++)
63  r3dized_poly_faces[iface][ivrt] = matpoly_faces[ivrt];
64  }
65 
66  r3d_init_poly(&r3dpoly, r3dized_poly_vrts, nvrts, r3dized_poly_faces, nface_vrts, nfaces);
67 
68  delete [] r3dized_poly_vrts;
69  delete [] nface_vrts;
70  for (int iface = 0; iface < nfaces; iface++)
71  delete [] r3dized_poly_faces[iface];
72  delete [] r3dized_poly_faces;
73 }

83 r3dpoly_to_matpolys(r3d_poly& r3dpoly,
84  std::vector< MatPoly<3> >& mat_polys) {
85  r3d_brep* poly_brep;
86  r3d_int ncomponents;
87  r3d_init_brep(&r3dpoly, &poly_brep, &ncomponents);
88 
89  mat_polys.clear();
90  if (ncomponents == 0) {
91  r3d_free_brep(&poly_brep, 0);
92  return;
93  }
94 
95  mat_polys.resize(ncomponents);
96  for (int ipoly = 0; ipoly < ncomponents; ipoly++) {
97  int nvrts = poly_brep[ipoly].numvertices;
98  std::vector<Point3> curpoly_vrts;
99  curpoly_vrts.reserve(nvrts);
100  // We only store unique vertices, so we need a map from r3d node indices
101  // to MatPoly node indices for when we process faces
102  std::vector<int> r3d2matpoly_vrt_ids(nvrts, -1);
103  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
104  Point3 cur_vrt;
105  for (int ixyz = 0; ixyz < 3; ixyz++)
106  cur_vrt[ixyz] = poly_brep[ipoly].vertices[ivrt].xyz[ixyz];
107  // Check if this point is already stored
108  for (int i = 0; i < curpoly_vrts.size(); i++)
109  if (cur_vrt == curpoly_vrts[i]) {
110  r3d2matpoly_vrt_ids[ivrt] = i;
111  break;
112  }
113  // If the point is unique, we store it
114  if (r3d2matpoly_vrt_ids[ivrt] == -1) {
115  r3d2matpoly_vrt_ids[ivrt] = curpoly_vrts.size();
116  curpoly_vrts.push_back(cur_vrt);
117  }
118  }
119  curpoly_vrts.shrink_to_fit();
120 
121  int nfaces = poly_brep[ipoly].numfaces;
122  std::vector< std::vector<int> > curpoly_face(nfaces);
123  for (int iface = 0; iface < nfaces; iface++) {
124  int face_nverts = poly_brep[ipoly].numvertsperface[iface];
125  for (int ifv = 0; ifv < face_nverts; ifv++) {
126  int cur_vrt_id = r3d2matpoly_vrt_ids[poly_brep[ipoly].faceinds[iface][ifv]];
127  // We only add unique node indices to the list of face's nodes
128  if (std::find(curpoly_face[iface].begin(), curpoly_face[iface].end(),
129  cur_vrt_id) == curpoly_face[iface].end())
130  curpoly_face[iface].push_back(cur_vrt_id);
131  }
132  }
133 
134  // Filter out degenerate faces
135  int ind_face = 0;
136  while (ind_face < curpoly_face.size())
137  if (curpoly_face[ind_face].size() > 2) ind_face++;
138  else curpoly_face.erase(curpoly_face.begin() + ind_face);
139 
140  mat_polys[ipoly].initialize(curpoly_vrts, curpoly_face);
141  }
142 
143  r3d_free_brep(&poly_brep, ncomponents);
144 }

156 void
157 split_convex_matpoly_r3d(const MatPoly<3>& mat_poly,
158  const Plane_t<3>& cutting_plane,
159  MatPoly<3>& lower_halfspace_poly,
160  MatPoly<3>& upper_halfspace_poly,
161  std::vector<double>& lower_halfspace_moments,
162  std::vector<double>& upper_halfspace_moments) {
163  //Translate the cutting plane to R3D format
164  r3d_plane r3d_cut_plane;
165  for (int ixyz = 0; ixyz < 3; ixyz++)
166  r3d_cut_plane.n.xyz[ixyz] = cutting_plane.normal[ixyz];
167  r3d_cut_plane.d = cutting_plane.dist2origin;
168 
169  r3d_poly r3dized_poly;
170  r3d_poly r3d_subpolys[2];
171  matpoly_to_r3dpoly(mat_poly, r3dized_poly);
172  r3d_split(&r3dized_poly, 1, r3d_cut_plane, &r3d_subpolys[1], &r3d_subpolys[0]);
173 
174  MatPoly<3>* subpoly_ptrs[2] = {&lower_halfspace_poly, &upper_halfspace_poly};
175  std::vector<double>* subpoly_moments_ptrs[2] = {&lower_halfspace_moments,
176  &upper_halfspace_moments};
177 
178  const int POLY_ORDER = 1;
179  r3d_real r3d_moments[R3D_NUM_MOMENTS(POLY_ORDER)];
180  for (int isp = 0; isp < 2; isp++) {
181  //Check if the subpoly is empty
182  if (r3d_subpolys[isp].nverts == 0) {
183  subpoly_ptrs[isp]->clear();
184  subpoly_moments_ptrs[isp]->clear();
185  continue;
186  }
187 
188  //Find the moments for a subpoly
189  r3d_reduce(&r3d_subpolys[isp], r3d_moments, POLY_ORDER);
190 //*
191  if (r3d_moments[0] <= std::numeric_limits<double>::epsilon()) {
192  subpoly_ptrs[isp]->clear();
193  subpoly_moments_ptrs[isp]->clear();
194  continue;
195  }
196 //*/
197  subpoly_moments_ptrs[isp]->assign(r3d_moments, r3d_moments + 4);
198 
199  //Get a MatPoly for a subpoly
200  std::vector< MatPoly<3> > sub_matpoly;
201  r3dpoly_to_matpolys(r3d_subpolys[isp], sub_matpoly);
202  int ncomponents = (int) sub_matpoly.size();
203  if (ncomponents > 1)
204  throw std::runtime_error("Non-convex MatPoly is split using the method for convex MatPoly's!");
205  if (ncomponents == 1)
206  *subpoly_ptrs[isp] = sub_matpoly[0];
207  }

254

257 
258  for (int ihs = 0; ihs < 2; ihs++)
259  hs_moments_ptrs[ihs]->assign(4, 0.0);

262  int npolys = (int) convex_polys->size();
263 
264  for (int icp = 0; icp < npolys; icp++) {
265  MatPoly<3> cur_subpolys[2];
266  std::vector<double> cur_moments[2];

268  cur_subpolys[0], cur_subpolys[1],
269  cur_moments[0], cur_moments[1]);
270 
271  for (int ihs = 0; ihs < 2; ihs++)
272  if (!cur_moments[ihs].empty()) {
273  hs_subpolys_ptrs[ihs]->emplace_back(cur_subpolys[ihs]);
274  for (int im = 0; im < 4; im++)

276  hs_poly_count[ihs]++;
277  }
278  }
279 
280  for (int ihs = 0; ihs < 2; ihs++)
281  if (hs_poly_count[ihs] == 0)
282  hs_moments_ptrs[ihs]->clear();
283 
284  return hs_sets;
285  }

307 void
308 lower_halfspace_moments_r3d(const MatPoly<3>& mat_poly,
309  const Plane_t<3>& cutting_plane,
310  std::vector<double>& lower_halfspace_moments) {
311  //Translate the cutting plane to R3D format
312  r3d_plane r3d_cut_plane;
313  for (int ixyz = 0; ixyz < 3; ixyz++)
314  r3d_cut_plane.n.xyz[ixyz] = -cutting_plane.normal[ixyz];
315  r3d_cut_plane.d = -cutting_plane.dist2origin;
316 
317  r3d_poly r3dized_poly;
318  matpoly_to_r3dpoly(mat_poly, r3dized_poly);
319  r3d_clip(&r3dized_poly, &r3d_cut_plane, 1);
320 
321  const int POLY_ORDER = 1;
322  r3d_real r3d_moments[R3D_NUM_MOMENTS(POLY_ORDER)];
323  //Check if the subpoly is empty
324  if (r3dized_poly.nverts == 0)
325  lower_halfspace_moments.clear();
326  else {
327  //Find the moments for a halfspace
328  r3d_reduce(&r3dized_poly, r3d_moments, POLY_ORDER);
329  lower_halfspace_moments.assign(r3d_moments, r3d_moments + 4);
330  }

367 
368  int poly_counter = 0;
369  if (planar_faces_)
370  for (int ipoly = 0; ipoly < matpolys_.size(); ipoly++) {
371  std::vector<double> cur_moments;

391  below_plane_moments[im] += cur_moments[im];
392  poly_counter++;
393  }
394  }
395 
396  if (poly_counter == 0)
397  below_plane_moments.clear();
398

400  }
