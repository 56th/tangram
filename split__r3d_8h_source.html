41 matpoly_to_r3dpoly(const MatPoly<3>& mat_poly,
42  r3d_poly& r3dpoly) {
43  //Translate coordinates of vertices to R3D format
44  const std::vector<Point3>& matpoly_vrts = mat_poly.points();
45  int nvrts = mat_poly.num_vertices();
46  r3d_rvec3* r3dized_poly_vrts;
47  r3dized_poly_vrts = new r3d_rvec3 [nvrts];
48  for (int ivrt = 0; ivrt < nvrts; ivrt++)
49  for (int ixyz = 0; ixyz < 3; ixyz++)
50  r3dized_poly_vrts[ivrt].xyz[ixyz] = matpoly_vrts[ivrt][ixyz];
51 
52  //Get faces of the polyhedron in terms of local indices of its vertices
53  int nfaces = mat_poly.num_faces();
54  r3d_int* nface_vrts;
55  nface_vrts = new r3d_int [nfaces];
56  r3d_int** r3dized_poly_faces;
57  r3dized_poly_faces = new r3d_int* [nfaces];
58  for (int iface = 0; iface < nfaces; iface++) {
59  const std::vector<int>& matpoly_faces = mat_poly.face_vertices(iface);
60  int face_nvrts = static_cast<int>(matpoly_faces.size());
61  nface_vrts[iface] = face_nvrts;
62  r3dized_poly_faces[iface] = new r3d_int [face_nvrts];
63  for (int ivrt = 0; ivrt < face_nvrts; ivrt++)
64  r3dized_poly_faces[iface][ivrt] = matpoly_faces[ivrt];
65  }
66 
67  r3d_init_poly(&r3dpoly, r3dized_poly_vrts, nvrts, r3dized_poly_faces, nface_vrts, nfaces);
68 
69  delete [] r3dized_poly_vrts;
70  delete [] nface_vrts;
71  for (int iface = 0; iface < nfaces; iface++)
72  delete [] r3dized_poly_faces[iface];
73  delete [] r3dized_poly_faces;

84 r3dpoly_to_matpolys(const r3d_poly& r3dpoly,
85  std::vector< MatPoly<3> >& mat_polys) {
86  r3d_brep* poly_brep;
87  r3d_int ncomponents;
88  r3d_init_brep(&r3dpoly, &poly_brep, &ncomponents);
89 
90  mat_polys.clear();
91  if (ncomponents == 0) {
92  r3d_free_brep(&poly_brep, 0);
93  return;
94  }
95 
96  mat_polys.reserve(ncomponents);
97  for (int ipoly = 0; ipoly < ncomponents; ipoly++) {
98  int nvrts = poly_brep[ipoly].numvertices;
99  std::vector<Point3> curpoly_vrts;
100  curpoly_vrts.reserve(nvrts);
101  // We only store unique vertices, so we need a map from r3d node indices
102  // to MatPoly node indices for when we process faces
103  std::vector<int> r3d2matpoly_vrt_ids(nvrts, -1);
104  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
105  Point3 cur_vrt;
106  for (int ixyz = 0; ixyz < 3; ixyz++)
107  cur_vrt[ixyz] = poly_brep[ipoly].vertices[ivrt].xyz[ixyz];
108  // Check if this point is already stored
109  for (int i = 0; i < curpoly_vrts.size(); i++)
110  if (cur_vrt == curpoly_vrts[i]) {
111  r3d2matpoly_vrt_ids[ivrt] = i;
112  break;
113  }
114  // If the point is unique, we store it
115  if (r3d2matpoly_vrt_ids[ivrt] == -1) {
116  r3d2matpoly_vrt_ids[ivrt] = curpoly_vrts.size();
117  curpoly_vrts.push_back(cur_vrt);
118  }
119  }
120  curpoly_vrts.shrink_to_fit();
121 
122  if (curpoly_vrts.size() < 4) continue;
123 
124  int nfaces = poly_brep[ipoly].numfaces;
125  std::vector< std::vector<int> > curpoly_faces(nfaces);
126  for (int iface = 0; iface < nfaces; iface++) {
127  int face_nverts = poly_brep[ipoly].numvertsperface[iface];
128  for (int ifv = 0; ifv < face_nverts; ifv++) {
129  int cur_vrt_id = r3d2matpoly_vrt_ids[poly_brep[ipoly].faceinds[iface][ifv]];
130  // We only add unique node indices to the list of face's nodes
131  if (std::find(curpoly_faces[iface].begin(), curpoly_faces[iface].end(),
132  cur_vrt_id) == curpoly_faces[iface].end())
133  curpoly_faces[iface].push_back(cur_vrt_id);
134  }
135  }
136 
137  // Filter out degenerate faces
138  int ind_face = 0;
139  while (ind_face < curpoly_faces.size())
140  if (curpoly_faces[ind_face].size() > 2) ind_face++;
141  else curpoly_faces.erase(curpoly_faces.begin() + ind_face);
142 
143  // We do not store polyhedra with less than four faces,
144  // as they are clearly degenerate
145  if (curpoly_faces.size() > 3) {
146  int inew_poly = static_cast<int>(mat_polys.size());
147  mat_polys.push_back(MatPoly<3>());
148  mat_polys[inew_poly].initialize(curpoly_vrts, curpoly_faces);
149  }
150  }
151  mat_polys.shrink_to_fit();
152 
153  r3d_free_brep(&poly_brep, ncomponents);
154 }

167 split_convex_matpoly_r3d(const MatPoly<3>& mat_poly,
168  const Plane_t<3>& cutting_plane,
169  MatPoly<3>& lower_halfspace_poly,
170  MatPoly<3>& upper_halfspace_poly,
171  std::vector<double>& lower_halfspace_moments,
172  std::vector<double>& upper_halfspace_moments) {
173  //Translate the cutting plane to R3D format
174  r3d_plane r3d_cut_plane;
175  for (int ixyz = 0; ixyz < 3; ixyz++)
176  r3d_cut_plane.n.xyz[ixyz] = cutting_plane.normal[ixyz];
177  r3d_cut_plane.d = cutting_plane.dist2origin;
178 
179  r3d_poly r3dized_poly;
180  r3d_poly r3d_subpolys[2];
181  matpoly_to_r3dpoly(mat_poly, r3dized_poly);
182  r3d_split(&r3dized_poly, 1, r3d_cut_plane, &r3d_subpolys[1], &r3d_subpolys[0]);
183 
184  MatPoly<3>* subpoly_ptrs[2] = {&lower_halfspace_poly, &upper_halfspace_poly};
185  std::vector<double>* subpoly_moments_ptrs[2] = {&lower_halfspace_moments,
186  &upper_halfspace_moments};
187 
188  const int POLY_ORDER = 1;
189  r3d_real r3d_moments[R3D_NUM_MOMENTS(POLY_ORDER)];
190  for (int isp = 0; isp < 2; isp++) {
191  //Check if the subpoly is empty
192  if (r3d_subpolys[isp].nverts == 0) {
193  subpoly_ptrs[isp]->clear();
194  subpoly_moments_ptrs[isp]->clear();
195  continue;
196  }
197 
198  //Find the moments for a subpoly
199  r3d_reduce(&r3d_subpolys[isp], r3d_moments, POLY_ORDER);
200  if (r3d_moments[0] <= std::numeric_limits<double>::epsilon()) {
201  subpoly_ptrs[isp]->clear();
202  subpoly_moments_ptrs[isp]->clear();
203  continue;
204  }
205  subpoly_moments_ptrs[isp]->assign(r3d_moments, r3d_moments + 4);
206 
207  //Get a MatPoly for a subpoly
208  std::vector< MatPoly<3> > sub_matpoly;
209  r3dpoly_to_matpolys(r3d_subpolys[isp], sub_matpoly);
210  int ncomponents = static_cast<int>(sub_matpoly.size());
211  if (ncomponents > 1) {
212  // Filter out degenerate components
213  int ind_subpoly = 0;
214  while (ind_subpoly < sub_matpoly.size())
215  if (sub_matpoly[ind_subpoly].moments()[0] >
216  std::numeric_limits<double>::epsilon()) ind_subpoly++;
217  else sub_matpoly.erase(sub_matpoly.begin() + ind_subpoly);
218  }
219 
220  if (ncomponents == 1)
221  *subpoly_ptrs[isp] = sub_matpoly[0];
222  else
223  throw std::runtime_error("Non-convex MatPoly is split using the method for convex MatPoly's!");
224  }

454 void get_intersection_moments(const MatPoly<3>& mat_poly,
455  const r3d_poly& r3dpoly,
456  std::vector<double>& intersection_moments,
457  bool convex_matpoly = false) {
458  const int POLY_ORDER = 1;
459  int nmoments = R3D_NUM_MOMENTS(POLY_ORDER);
460  r3d_real r3d_moments[R3D_NUM_MOMENTS(POLY_ORDER)];
461 
462  if (convex_matpoly) {
463  std::vector< Plane_t<3> > face_planes;
464  mat_poly.face_planes(face_planes);
465  r3d_poly intersection = r3dpoly;
466 
467  int nplanes = static_cast<int>(face_planes.size());
468  r3d_plane* r3d_face_planes = new r3d_plane [nplanes];
469  for (int iplane = 0; iplane < nplanes; iplane++) {
470  for (int ixyz = 0; ixyz < 3; ixyz++)
471  r3d_face_planes[iplane].n.xyz[ixyz] = -face_planes[iplane].normal[ixyz];
472  r3d_face_planes[iplane].d = -face_planes[iplane].dist2origin;
473  }
474 
475  r3d_clip(&intersection, r3d_face_planes, (r3d_int) nplanes);
476 
477  delete [] r3d_face_planes;
478 
479  r3d_reduce(&intersection, r3d_moments, POLY_ORDER);
480  intersection_moments.resize(nmoments);
481  for (int im = 0; im < nmoments; im++)
482  intersection_moments[im] = r3d_moments[im];
483  }
484  else {
485  std::vector< MatPoly<3> > mat_poly_tets;
486  mat_poly.facetize_decompose(mat_poly_tets);
487 
488  int ntets = static_cast<int>(mat_poly_tets.size());
489  intersection_moments.assign(nmoments, 0.0);
490  for (int itet = 0; itet < ntets; itet++) {
491  std::vector< Plane_t<3> > face_planes;
492  mat_poly_tets[itet].face_planes(face_planes);
493  assert(face_planes.size() == 4);
494 
495  r3d_poly intersection = r3dpoly;
496  r3d_plane r3d_face_planes[4];
497  for (int iplane = 0; iplane < 4; iplane++) {
498  for (int ixyz = 0; ixyz < 3; ixyz++)
499  r3d_face_planes[iplane].n.xyz[ixyz] = -face_planes[iplane].normal[ixyz];
500  r3d_face_planes[iplane].d = -face_planes[iplane].dist2origin;
501  }
502 
503  r3d_clip(&intersection, r3d_face_planes, 4);
504 
505  r3d_reduce(&intersection, r3d_moments, POLY_ORDER);
506  for (int im = 0; im < nmoments; im++)
507  intersection_moments[im] += r3d_moments[im];
508  }
509  }
