33 double quad_interpolant_minimizer(const double arg0,
34  const double arg1,
35  const double fval0,
36  const double fval1,
37  const double df0,
38  const double* safeguards) {
39  double darg = arg1 - arg0;
40  double denominator = fval1 - fval0 - df0*(darg);
41  double arg_min;
42  double bnd0 = arg0 + safeguards[0]*darg,
43  bnd1 = arg0 + safeguards[1]*darg;
44 
45  if (std::fabs(denominator) <= std::numeric_limits<double>::epsilon())
46  arg_min = (fval1 > fval0) ? bnd0 : bnd1;
47  else {
48  arg_min = arg0 - df0*pow2(darg)/(2*denominator);
49  double d2arg0 = std::fabs(arg_min - arg0);
50  if (d2arg0 < std::fabs(bnd0 - arg0)) arg_min = bnd0;
51  else if (d2arg0 > std::fabs(bnd1 - arg0)) arg_min = bnd1;
52  }
53 
54  return arg_min;
55 }

66 template<int arg_dim>
67 void fwd_diff_grad(const std::function<double(const Vector<arg_dim>&)>& fun,
68  const Vector<arg_dim>& arg_val,
69  const double& fun_val,
70  Vector<arg_dim>& approx_grad,
71  double& fdiff_h) {
72  fdiff_h = sqrt(std::numeric_limits<double>::epsilon());
73  Vector<arg_dim> fwd_arg = arg_val;
74  for (int i = 0; i < arg_dim; i++) {
75  fwd_arg[i] += fdiff_h;
76  double forward_fval = fun(fwd_arg);
77  fwd_arg[i] = arg_val[i];
78 
79  approx_grad[i] = (forward_fval - fun_val)/fdiff_h;
80  }
81 }

99 template<int arg_dim>
100 double linesearch(const std::function<double(const Vector<arg_dim>&)>& obj_fun,
101  const double& obj_fun_lbnd,
102  const Vector<arg_dim>& arg,
103  const Vector<arg_dim>& dir,
104  const double& initial_guess,
105  const double& c1,
106  const double& c2,
107  const IterativeMethodTolerances_t& im_tols,
108  double& cur_fval,
109  Vector<arg_dim>& cur_grad) {
110  // Bracketing step, solution will be in (alpha_bnd[0], alpha_bnd[1])
111  double fval0 = cur_fval;
112  double df0 = dot(cur_grad, dir);
113  assert(df0 < -std::numeric_limits<double>::epsilon());
114  double fdiff_h;
115 
116  auto fval = [&obj_fun, &arg, &dir](double alpha) {
117  return obj_fun(arg + alpha*dir);
118  };
119 
120  // alpha_max is chosen so that it should violate the sufficient decrease condition
121  // and therefore provides an acceptable upper bound for the interval
122  double alpha_max = (obj_fun_lbnd - fval0)/(c1*df0);
123 
124  double alpha_bnd[2] = {0.0, initial_guess};
125  double bnd_fval[2]; bnd_fval[0] = fval0;
126  double df_lbnd = df0;
127  // At the beginning of each loop, cur_fval and cur_grad are at alpha_bnd[0]
128  for (int i = 0; i < im_tols.max_num_iter; i++) {
129  bnd_fval[1] = fval(alpha_bnd[1]);
130 
131  if (bnd_fval[1] - obj_fun_lbnd <= im_tols.fun_eps) {
132  cur_fval = bnd_fval[1];
133  fwd_diff_grad<arg_dim>(obj_fun, arg + alpha_bnd[1]*dir, bnd_fval[1], cur_grad, fdiff_h);
134  return alpha_bnd[1];
135  }
136 
137  if ( (bnd_fval[1] > fval0 + c1*alpha_bnd[1]*df0) ||
138  ((bnd_fval[1] >= bnd_fval[0]) && (i > 0)) )
139  break;
140 
141  // alpha_bnd[1] will be the next alpha_bnd[0]
142  cur_fval = bnd_fval[1];
143  fwd_diff_grad<arg_dim>(obj_fun, arg + alpha_bnd[1]*dir, bnd_fval[1], cur_grad, fdiff_h);
144  df_lbnd = dot(cur_grad, dir);
145 
146  if (std::fabs(df_lbnd) <= -c2*df0)
147  return alpha_bnd[1];
148 
149  if (df_lbnd >= 0.0) {
150  std::reverse(std::begin(alpha_bnd), std::end(alpha_bnd));
151  std::reverse(std::begin(bnd_fval), std::end(bnd_fval));
152  break;
153  }
154 
155  double dalpha = alpha_bnd[1] - alpha_bnd[0];
156  double tau = 9.0; // interval length scaling
157 
158  alpha_bnd[0] = alpha_bnd[1];
159  bnd_fval[0] = bnd_fval[1];
160  alpha_bnd[1] += tau*dalpha;
161 
162  if (alpha_bnd[1] > alpha_max - im_tols.arg_eps) {
163  alpha_bnd[1] = alpha_max;
164  bnd_fval[1] = obj_fun(alpha_max);
165  break;
166  }
167  }
168 
169  // Sectioning/zooming step
170  // At the beginning of each loop, cur_fval and cur_grad are at alpha_bnd[0]
171  double safeguards[2] = {0.1, 0.5};
172  for (int i = 0; i < im_tols.max_num_iter; i++) {
173  if (std::fabs(alpha_bnd[1] - alpha_bnd[0]) < im_tols.arg_eps)
174  return alpha_bnd[0];
175 
176  double cur_alpha = quad_interpolant_minimizer(alpha_bnd[0], alpha_bnd[1], bnd_fval[0],
177  bnd_fval[1], df_lbnd, safeguards);
178 
179  cur_fval = fval(cur_alpha);
180  if (cur_fval - obj_fun_lbnd <= im_tols.fun_eps) {
181  fwd_diff_grad<arg_dim>(obj_fun, arg + cur_alpha*dir, cur_fval, cur_grad, fdiff_h);
182  return cur_alpha;
183  }
184 
185  if ( (cur_fval > fval0 + c1*cur_alpha*df0) || (cur_fval >= bnd_fval[0]) ) {
186  alpha_bnd[1] = cur_alpha;
187  bnd_fval[1] = cur_fval;
188  cur_fval = bnd_fval[0];
189  continue;
190  }
191 
192  fwd_diff_grad<arg_dim>(obj_fun, arg + cur_alpha*dir, cur_fval, cur_grad, fdiff_h);
193  double cur_df = dot(cur_grad, dir);
194  if (std::fabs(cur_df) <= -c2*df0)
195  return cur_alpha;
196 
197  if (cur_df*(alpha_bnd[1] - alpha_bnd[0]) >= 0) {
198  alpha_bnd[1] = alpha_bnd[0];
199  bnd_fval[1] = bnd_fval[0];
200  }
201 
202  alpha_bnd[0] = cur_alpha;
203  bnd_fval[0] = cur_fval;
204  df_lbnd = cur_df;
205  }
206 
207  return alpha_bnd[0];
208 }

231 template<int arg_dim>
232 double mwwp_linesearch(const std::function<double(const Vector<arg_dim>&)>& obj_fun,
233  const double& obj_fun_lbnd,
234  const Vector<arg_dim>& arg,
235  const Vector<arg_dim>& dir,
236  const double initial_guess,
237  const double delta,
238  const double delta1,
239  const IterativeMethodTolerances_t& im_tols,
240  double& cur_fval,
241  Vector<arg_dim>& cur_grad) {
242  // Bracketing step, solution will be in (alpha_bnd[0], alpha_bnd[1])
243  double phi0 = cur_fval;
244  double dir_dot_dir = dir.norm(false);
245 
246  double dphi0 = dot(cur_grad, dir);
247  assert(dphi0 < -std::numeric_limits<double>::epsilon());
248  double fdiff_h;
249 
250  auto fval = [&obj_fun, &arg, &dir](double alpha) {
251  return obj_fun(arg + alpha*dir);
252  };
253 
254  auto phi = [&delta, &delta1, &phi0, &dphi0, &dir_dot_dir](double alpha, double f_alpha) {
255  return f_alpha - phi0 - delta*dphi0*alpha -
256  alpha*std::min(-delta1*dphi0, 0.5*delta*dir_dot_dir*alpha);
257  };
258 
259  auto dphi = [&dir, &delta, &delta1, &dphi0, &dir_dot_dir](
260  double alpha, Vector<arg_dim>& grad_alpha) {
261  return dot(grad_alpha, dir) - delta*dphi0 -
262  std::min(-delta1*dphi0, delta*dir_dot_dir*alpha);
263  };
264 
265  double alpha_max = (obj_fun_lbnd - phi0)/((delta - delta1)*dphi0);
266 
267  double alpha_bnd[2] = {0.0, initial_guess};
268  double bnd_phi[2]; bnd_phi[0] = phi0;
269  double dphi_lbnd = dphi0;
270 
271  // At the beginning of each loop, cur_fval and cur_grad are at alpha_bnd[0]
272  for (int i = 0; i < im_tols.max_num_iter; i++) {
273  double ubnd_fval = fval(alpha_bnd[1]);
274  if (ubnd_fval - obj_fun_lbnd <= im_tols.fun_eps) {
275  cur_fval = ubnd_fval;
276  fwd_diff_grad<arg_dim>(obj_fun, arg + alpha_bnd[1]*dir, ubnd_fval, cur_grad, fdiff_h);
277  return alpha_bnd[1];
278  }
279 
280  bnd_phi[1] = phi(alpha_bnd[1], ubnd_fval);
281  if (bnd_phi[1] > 0.0)
282  break;
283 
284  cur_fval = ubnd_fval;
285  fwd_diff_grad<arg_dim>(obj_fun, arg + alpha_bnd[1]*dir, ubnd_fval, cur_grad, fdiff_h);
286  dphi_lbnd = dphi(alpha_bnd[1], cur_grad);
287 
288  if (dphi_lbnd >= 0.0)
289  return alpha_bnd[1];
290 
291  if (alpha_bnd[1] > alpha_max - im_tols.arg_eps)
292  return alpha_bnd[1];
293 
294 
295  double dalpha = alpha_bnd[1] - alpha_bnd[0];
296  double tau = 9.0; // interval length scaling
297 
298  alpha_bnd[0] = alpha_bnd[1];
299  bnd_phi[0] = bnd_phi[1];
300  alpha_bnd[1] += tau*dalpha;
301 
302  if (alpha_bnd[1] > alpha_max - im_tols.arg_eps) {
303  alpha_bnd[1] = alpha_max;
304  double alpha_max_fval = obj_fun(alpha_max);
305  bnd_phi[1] = phi(alpha_max, alpha_max_fval);
306  break;
307  }
308  }
309 
310  // Sectioning/zooming step
311  // At the beginning of each loop, cur_fval and cur_grad are at alpha_bnd[0]
312  double lbnd_fval = cur_fval;
313  double safeguards[2] = {0.1, 0.5};
314  for (int i = 0; i < im_tols.max_num_iter; i++) {
315  if (std::fabs(alpha_bnd[1] - alpha_bnd[0]) < im_tols.arg_eps)
316  return alpha_bnd[0];
317 
318  double cur_alpha = quad_interpolant_minimizer(alpha_bnd[0], alpha_bnd[1], bnd_phi[0],
319  bnd_phi[1], dphi_lbnd, safeguards);
320 
321  cur_fval = fval(cur_alpha);
322  if (cur_fval - obj_fun_lbnd <= im_tols.fun_eps) {
323  fwd_diff_grad<arg_dim>(obj_fun, arg + cur_alpha*dir, cur_fval, cur_grad, fdiff_h);
324  return cur_alpha;
325  }
326 
327  double cur_phi = phi(cur_alpha, cur_fval);
328  if ( (cur_phi > 0.0) || (cur_phi >= bnd_phi[0]) ) {
329  alpha_bnd[1] = cur_alpha;
330  bnd_phi[1] = cur_phi;
331  cur_fval = lbnd_fval;
332  continue;
333  }
334  fwd_diff_grad<arg_dim>(obj_fun, arg + cur_alpha*dir, cur_fval, cur_grad, fdiff_h);
335  double cur_dphi = dphi(cur_alpha, cur_grad);
336  if (cur_dphi >= 0.0)
337  return cur_alpha;
338 
339  alpha_bnd[0] = cur_alpha;
340  bnd_phi[0] = cur_phi;
341  lbnd_fval = cur_fval;
342  dphi_lbnd = cur_dphi;
343  }
344 
345  return alpha_bnd[0];
346 }

356 template<int arg_dim>
357 Vector<arg_dim> bfgs(const std::function<double(const Vector<arg_dim>&)>& obj_fun,
358  const double& obj_fun_lbnd,
359  const Vector<arg_dim>& init_guess,
360  const IterativeMethodTolerances_t& im_tols) {
361  const double linesearch_c1 = 1.0e-4;
362  const double linesearch_c2 = 0.9;
363  const double damping_scalar = 0.2;
364 
365  double fdiff_h;
366 
367  //Initial Hessian approximation: identity matrix
368  Matrix B(arg_dim, arg_dim, 0.0);
369  for (int i = 0; i < arg_dim; i++)
370  B[i][i] = 1.0;
371 
372  Vector<arg_dim> cur_arg = init_guess, prev_grad, cur_grad, cur_dir, darg, dgrad;
373  double dfval, cur_fval = obj_fun(cur_arg);
374  fwd_diff_grad<arg_dim>(obj_fun, cur_arg, cur_fval, cur_grad, fdiff_h);
375 
376  for (int i = 0; i < im_tols.max_num_iter; i++) {
377  // Forward differences are O(fdiff_h) accurate
378  if (cur_grad.max_norm()*fdiff_h/std::max(1.0, std::fabs(cur_fval)) <=
379  std::numeric_limits<double>::epsilon())
380  break;
381 
382  Wonton::solve<arg_dim>(B, -cur_grad, cur_dir);
383 
384  double df0 = dot(cur_grad, cur_dir);
385  // Function is flat in the linesearch direction
386  if (df0 >= -std::numeric_limits<double>::epsilon())
387  break;
388 
389  double initial_alpha = 1.0;
390  if (i > 0) initial_alpha = std::min(1.0,
391  -2*std::max(-dfval, 10*std::numeric_limits<double>::epsilon())/df0);
392 
393  dfval = -cur_fval;
394  prev_grad = cur_grad;
395 
396  double alpha = linesearch<arg_dim>(obj_fun, obj_fun_lbnd,
397  cur_arg, cur_dir, initial_alpha,
398  linesearch_c1, linesearch_c2,
399  im_tols, cur_fval, cur_grad);
400  dfval += cur_fval;
401  darg = alpha*cur_dir;
402 
403  if (darg.norm() < im_tols.arg_eps)
404  break;
405 
406  cur_arg += darg;
407  if (std::fabs(cur_fval) <= im_tols.fun_eps)
408  return cur_arg;
409 
410  dgrad = cur_grad - prev_grad;
411 
412  // We update B if the change in gradient is above
413  // forward differences error
414  if (dgrad.max_norm() > fdiff_h) {
415  Vector<arg_dim> B_darg = B*darg;
416  if (B_darg.max_norm() <= fdiff_h)
417  continue;
418 
419  double darg_B_darg = dot(darg, B_darg);
420  double darg_dgrad = dot(darg, dgrad);
421 
422  // Perform damping if necessary
423  if (darg_dgrad < damping_scalar*darg_B_darg) {
424  double delta = (1.0 - damping_scalar)*darg_B_darg/(darg_B_darg - darg_dgrad);
425  dgrad *= delta;
426  dgrad += (1.0 - delta)*B_darg;
427  darg_dgrad = dot(darg, dgrad);
428  }
429 
430  if (darg_dgrad <= std::numeric_limits<double>::epsilon())
431  continue;
432 
433  // Correction to the initial Hessian guess
434  if (i == 0) {
435  B *= dgrad.norm(false)/darg_dgrad;
436  B_darg = B*darg;
437  darg_B_darg = dot(darg, B_darg);
438  }
439  B += (1.0/darg_dgrad)*(dgrad*dgrad) - (1.0/darg_B_darg)*(B_darg*B_darg);
440  }
441  }
442 
443  return cur_arg;
444 }

557 template<int arg_dim>
558 Vector<arg_dim> dbfgs(const std::function<double(const Vector<arg_dim>&)>& obj_fun,
559  const double& obj_fun_lbnd,
560  const Vector<arg_dim>& init_guess,
561  const IterativeMethodTolerances_t& im_tols) {
562  const double linesearch_c1 = 1.0e-4;
563  const double linesearch_c2 = 0.9;
564 
565  double fdiff_h;
566 
567  //Initial Hessian approximation: identity matrix
568  Matrix B(arg_dim, arg_dim, 0.0);
569  for (int i = 0; i < arg_dim; i++)
570  B[i][i] = 1.0;
571 
572  Vector<arg_dim> cur_arg = init_guess, prev_grad, cur_grad, cur_dir, darg, dgrad;
573  double dfval, cur_fval = obj_fun(cur_arg);
574  fwd_diff_grad<arg_dim>(obj_fun, cur_arg, cur_fval, cur_grad, fdiff_h);
575 
576  for (int i = 0; i < im_tols.max_num_iter; i++) {
577  // Forward differences are O(fdiff_h) accurate
578  if (cur_grad.max_norm()*fdiff_h/std::max(1.0, std::fabs(cur_fval)) <=
579  std::numeric_limits<double>::epsilon())
580  break;
581 
582  Wonton::solve<arg_dim>(B, -cur_grad, cur_dir);
583 
584  double df0 = dot(cur_grad, cur_dir);
585  // Function is flat in the linesearch direction
586  if (df0 >= -std::numeric_limits<double>::epsilon())
587  break;
588 
589  double initial_alpha = 1.0;
590  if (i > 0) initial_alpha = std::min(1.0,
591  -2*std::max(-dfval, 10*std::numeric_limits<double>::epsilon())/df0);
592 
593  dfval = -cur_fval;
594  prev_grad = cur_grad;
595 
596  double alpha = linesearch<arg_dim>(obj_fun, obj_fun_lbnd,
597  cur_arg, cur_dir, initial_alpha,
598  linesearch_c1, linesearch_c2,
599  im_tols, cur_fval, cur_grad);
600  dfval += cur_fval;
601  darg = alpha*cur_dir;
602  if (darg.norm() < im_tols.arg_eps)
603  break;
604 
605  cur_arg += darg;
606  if (std::fabs(cur_fval) <= im_tols.fun_eps)
607  return cur_arg;
608 
609  dgrad = cur_grad - prev_grad;
610 
611  // We update B if the change in gradient is above
612  // forward differences error
613  if (dgrad.max_norm() > fdiff_h) {
614  Vector<arg_dim> B_darg = B*darg;
615  if (B_darg.max_norm() <= fdiff_h)
616  continue;
617 
618  double darg_B_darg = dot(darg, B_darg);
619  double darg_dgrad = dot(darg, dgrad);
620 
621  double b_rec = darg_dgrad/darg_B_darg;
622  Vector<arg_dim> H_dgrad;
623  Wonton::solve<arg_dim>(B, dgrad, H_dgrad);
624  double h_rec = darg_dgrad/dot(dgrad, H_dgrad);
625 
626  double l = std::min(b_rec, b_rec*h_rec);
627  double m = std::max(b_rec, b_rec*h_rec);
628 
629  double sigma2 = std::max(1.0 - 1.0/alpha, 0.5);
630  double sigma3 = exp(1);
631 
632  double phi = 1.0;
633  if (l < 1.0 - sigma2)
634  phi = sigma2/(1.0 - l);
635  else if (m > 1 + sigma3)
636  phi = sigma3/(m - 1.0);
637  else {
638  double a = 1.0/(b_rec*h_rec);
639  if ( (l >= 1.0 - sigma2) && (m <= 1.0 + sigma3) && (a > sigma3) )
640  phi = sqrt(sigma3/a);
641  }
642 
643  // Perform damping if necessary
644  if (phi != 1.0) {
645  dgrad *= phi;
646  dgrad += (1.0 - phi)*B_darg;
647  darg_dgrad = dot(darg, dgrad);
648  }
649 
650  if (darg_dgrad <= std::numeric_limits<double>::epsilon())
651  continue;
652 
653  // Correction to the initial Hessian guess
654  if (i == 0) {
655  B *= dgrad.norm(false)/darg_dgrad;
656  B_darg = B*darg;
657  darg_B_darg = dot(darg, B_darg);
658  }
659  B += (1.0/darg_dgrad)*(dgrad*dgrad) - (1.0/darg_B_darg)*(B_darg*B_darg);
660  }
661  }
662 
663  return cur_arg;
664 }
