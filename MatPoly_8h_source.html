293 template<>
294 Point3 MatPoly<3>::face_centroid(int const face_id) const {
295 #ifdef DEBUG
296  assert((face_id >= 0) && (face_id < nfaces_));
297 #endif
298  Point3 centroid;
299 
300  int nvrts = (int) face_vertices_[face_id].size();
301  if (nvrts == 3) {
302  for (int ivrt = 0; ivrt < 3; ivrt++)
303  centroid += vertex_points_[face_vertices_[face_id][ivrt]];
304  centroid /= 3.0;
305 
306  return centroid;
307  }
308 
309  Point3 gcenter;
310  for (int ivrt = 0; ivrt < nvrts; ivrt++)
311  gcenter += vertex_points_[face_vertices_[face_id][ivrt]];
312  gcenter /= nvrts;
313 
314  double size = 0.0;
315  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
316  int ifv = face_vertices_[face_id][ivrt];
317  int isv = face_vertices_[face_id][(ivrt + 1)%nvrts];
318  Vector3 vec0 = vertex_points_[isv] - vertex_points_[ifv];
319  Vector3 vec1 = gcenter - vertex_points_[ifv];
320  double tri_size = 0.5*cross(vec0, vec1).norm();
321  size += tri_size;
322  Point3 tri_centroid = (vertex_points_[ifv] + vertex_points_[isv] + gcenter)/3.0;
323  centroid += tri_size*tri_centroid;
324  }
325  if (size > std::numeric_limits<double>::epsilon()) centroid /= size;
326  else centroid = gcenter;
327 
328  return centroid;
329 }

381 void MatPoly<2>::compute_moments(std::vector<double>& moments) const {
382  moments.assign(3, 0.0);
383 
384  for (int ivrt = 0; ivrt < nvertices_; ivrt++) {
385  double cur_term = vertex_points_[ivrt][0]*vertex_points_[(ivrt + 1)%nvertices_][1] -
386  vertex_points_[ivrt][1]*vertex_points_[(ivrt + 1)%nvertices_][0];
387  moments[0] += cur_term;
388  for (int idim = 0; idim < 2; idim++)
389  moments[idim + 1] += cur_term*(
390  vertex_points_[ivrt][idim] + vertex_points_[(ivrt + 1)%nvertices_][idim]);
391  }
392  moments[0] /= 2.0;
393  for (int idim = 0; idim < 2; idim++)
394  moments[idim + 1] /= 6.0;

403 void MatPoly<3>::compute_moments(std::vector<double>& moments) const {
404  moments.assign(4, 0.0);
405 
406  for (int iface = 0; iface < nfaces_; iface++) {
407  std::vector<Point3> face_pts = face_points(iface);
408  std::vector< std::vector<int> > itri_pts;
409 
410  int nvrts = face_vertices_[iface].size();
411  if (nvrts == 3)
412  itri_pts.push_back({0, 1, 2});
413  else {
414  itri_pts.reserve(nvrts);
415  face_pts.emplace_back(face_centroid(iface));
416  for (int ivrt = 0; ivrt < nvrts; ivrt++)
417  itri_pts.push_back({nvrts, ivrt, (ivrt + 1)%nvrts});
418  }
419 
420  for (int itri = 0; itri < itri_pts.size(); itri++) {
421  Vector3 vcp = cross(face_pts[itri_pts[itri][1]] - face_pts[itri_pts[itri][0]],
422  face_pts[itri_pts[itri][2]] - face_pts[itri_pts[itri][0]]);
423  moments[0] += dot(vcp, face_pts[itri_pts[itri][0]].asV());
424  for (int idim = 0; idim < 3; idim++)
425  for (int ivrt = 0; ivrt < 3; ivrt++)
426  moments[idim + 1] += vcp[idim]*pow(face_pts[itri_pts[itri][ivrt]][idim] +
427  face_pts[itri_pts[itri][(ivrt + 1)%3]][idim], 2);
428  }
429  }
430 
431  moments[0] /= 6.0;
432  for (int idim = 0; idim < 3; idim++)
433  moments[idim + 1] /= 48.0;
434 }
