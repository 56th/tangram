20 double polygon3d_area(const std::vector<Point3>& points,
21  const std::vector<int>& poly_vertices) {
22  double poly_area = 0.0;
23 
24  int nvrts = static_cast<int>(poly_vertices.size());
25  if (nvrts == 3) {
26  poly_area = 0.5*cross(points[poly_vertices[1]] - points[poly_vertices[0]],
27  points[poly_vertices[2]] - points[poly_vertices[0]]).norm();
28 
29  if (std::fabs(poly_area) <= std::numeric_limits<double>::epsilon())
30  return 0.0;
31  }
32 
33  Point3 gcenter;
34  for (int ivrt = 0; ivrt < nvrts; ivrt++)
35  gcenter += points[poly_vertices[ivrt]];
36  gcenter /= nvrts;
37 
38  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
39  int ifv = poly_vertices[ivrt];
40  int isv = poly_vertices[(ivrt + 1)%nvrts];
41  double tri_size =
42  0.5*cross(points[isv] - points[ifv], gcenter - points[ifv]).norm();
43 
44  if (std::fabs(tri_size) > std::numeric_limits<double>::epsilon())
45  poly_area += tri_size;
46  }
47 
48  if (std::fabs(poly_area) <= std::numeric_limits<double>::epsilon())
49  return 0.0;
50 
51  return poly_area;
52 }

54 void polygon3d_moments(const std::vector<Point3>& points,
55  const std::vector<int>& poly_vertices,
56  std::vector<double>& poly_moments) {
57  poly_moments.assign(4, 0.0);
58 
59  int nvrts = static_cast<int>(poly_vertices.size());
60  if (nvrts == 3) {
61  poly_moments[0] = 0.5*cross(points[poly_vertices[1]] - points[poly_vertices[0]],
62  points[poly_vertices[2]] - points[poly_vertices[0]]).norm();
63 
64  if (std::fabs(poly_moments[0]) <= std::numeric_limits<double>::epsilon())
65  poly_moments[0] = 0.0;
66  else {
67  for (int ivrt = 0; ivrt < 3; ivrt++)
68  for (int idim = 0; idim < 3; idim++)
69  poly_moments[idim + 1] += points[poly_vertices[ivrt]][idim];
70  for (int idim = 0; idim < 3; idim++)
71  poly_moments[idim + 1] *= poly_moments[0]/3.0;
72  }
73 
74  return;
75  }
76 
77  Point3 gcenter;
78  for (int ivrt = 0; ivrt < nvrts; ivrt++)
79  gcenter += points[poly_vertices[ivrt]];
80  gcenter /= nvrts;
81 
82  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
83  int ifv = poly_vertices[ivrt];
84  int isv = poly_vertices[(ivrt + 1)%nvrts];
85  double tri_size =
86  0.5*cross(points[isv] - points[ifv], gcenter - points[ifv]).norm();
87 
88  if (std::fabs(tri_size) > std::numeric_limits<double>::epsilon()) {
89  poly_moments[0] += tri_size;
90  for (int idim = 0; idim < 3; idim++)
91  poly_moments[idim + 1] +=
92  tri_size*(points[ifv][idim] + points[isv][idim] + gcenter[idim])/3.0;
93  }
94  }
95 
96  if (std::fabs(poly_moments[0]) <= std::numeric_limits<double>::epsilon())
97  poly_moments.assign(4, 0.0);
98 }

101  const std::vector<int>& poly_vertices) {

103  polygon3d_moments(points, poly_vertices, poly_moments);
104 
105  Vector3 poly_normal(0.0, 0.0, 0.0);
106  if (poly_moments[0] <= std::numeric_limits<double>::epsilon())
107  return poly_normal;

111  poly_centroid[idim] = poly_moments[idim + 1]/poly_moments[0];

500 void MatPoly<2>::compute_moments(std::vector<double>& moments) const {
501  moments.assign(3, 0.0);
502 
503  for (int ivrt = 0; ivrt < nvertices_; ivrt++) {
504  double cur_term = vertex_points_[ivrt][0]*vertex_points_[(ivrt + 1)%nvertices_][1] -
505  vertex_points_[ivrt][1]*vertex_points_[(ivrt + 1)%nvertices_][0];
506  moments[0] += cur_term;
507  for (int idim = 0; idim < 2; idim++)
508  moments[idim + 1] += cur_term*(
509  vertex_points_[ivrt][idim] + vertex_points_[(ivrt + 1)%nvertices_][idim]);
510  }
511  moments[0] /= 2.0;
512  for (int idim = 0; idim < 2; idim++)
513  moments[idim + 1] /= 6.0;

521 template<>
522 void MatPoly<3>::compute_moments(std::vector<double>& moments) const {
523  moments.assign(4, 0.0);
524 
525  for (int iface = 0; iface < nfaces_; iface++) {
526  std::vector<double> face_moments;
527  polygon3d_moments(vertex_points_, face_vertices_[iface], face_moments);
528 
529  if (std::fabs(face_moments[0]) <= std::numeric_limits<double>::epsilon())
530  continue;
531 
532  std::vector<Point3> face_pts = face_points(iface);
533  std::vector< std::vector<int> > itri_pts;
534 
535  int nvrts = face_vertices_[iface].size();
536  if (nvrts == 3)
537  itri_pts.push_back({0, 1, 2});
538  else {
539  itri_pts.reserve(nvrts);
540  face_pts.push_back(Point3(face_moments[1]/face_moments[0],
541  face_moments[2]/face_moments[0], face_moments[3]/face_moments[0]));
542  for (int ivrt = 0; ivrt < nvrts; ivrt++)
543  itri_pts.push_back({nvrts, ivrt, (ivrt + 1)%nvrts});
544  }
545 
546  for (int itri = 0; itri < itri_pts.size(); itri++) {
547  Vector3 vcp = cross(face_pts[itri_pts[itri][1]] - face_pts[itri_pts[itri][0]],
548  face_pts[itri_pts[itri][2]] - face_pts[itri_pts[itri][0]]);
549  if (vcp.norm() <= std::numeric_limits<double>::epsilon())
550  continue;
551 
552  moments[0] += dot(vcp, face_pts[itri_pts[itri][0]].asV());
553  for (int idim = 0; idim < 3; idim++)
554  for (int ivrt = 0; ivrt < 3; ivrt++)
555  moments[idim + 1] += vcp[idim]*pow(face_pts[itri_pts[itri][ivrt]][idim] +
556  face_pts[itri_pts[itri][(ivrt + 1)%3]][idim], 2);
557  }
558  }
559 
560  moments[0] /= 6.0;
561  for (int idim = 0; idim < 3; idim++)
562  moments[idim + 1] /= 48.0;
563 }

814 bool point_inside_matpoly(const MatPoly<3> mat_poly,
815  const Point<3>& pt, bool convex_poly=false) {
816  std::vector< MatPoly<3> > convex_polys;
817  if (convex_poly)
818  convex_polys.push_back(mat_poly);
819  else
820  mat_poly.facetize_decompose(convex_polys);
821 
822  for (int icp = 0; icp < convex_polys.size(); icp++) {
823  const std::vector< Point<3> >& poly_pts = convex_polys[icp].points();
824 
825  bool pt_inside_cur_poly = true;
826  for (int iface = 0; iface < convex_polys[icp].num_faces(); iface++) {
827  const std::vector<int>& iface_vrts = convex_polys[icp].face_vertices(iface);
828  Vector3 pt2vrt_vec;
829  double pt2vrt_dst = 0.0;
830  //Find a face vertex that is the farthest from the given point
831  for (int ifvrt = 0; ifvrt < iface_vrts.size(); ifvrt++) {
832  Vector3 cur_pt2vrt_vec =
833  convex_polys[icp].vertex_point(iface_vrts[ifvrt]) - pt;
834  double cur_vec_norm = cur_pt2vrt_vec.norm();
835  if (cur_vec_norm > pt2vrt_dst) {
836  pt2vrt_vec = cur_pt2vrt_vec;
837  pt2vrt_dst = cur_vec_norm;
838  }
839  }
840  pt2vrt_vec /= pt2vrt_dst;
841 
842  Vector3 face_normal = polygon3d_normal(poly_pts, iface_vrts);
843  if (face_normal.is_zero())
844  continue;
845 
846  double fnormal_prj = dot(pt2vrt_vec, face_normal);
847  //Check the sign of the projection onto the normal
848  if (fnormal_prj <= std::numeric_limits<double>::epsilon()) {
849  pt_inside_cur_poly = false;
850  break;
851  }
852  }
853 
854  if (pt_inside_cur_poly)
855  return true;
856  }
857  return false;
858 }
