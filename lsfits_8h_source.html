48  // will only be nvals-1
49 
50  int nvals = vals.size();

52  // Each row of A contains the components of the vector from
53  // coord0 to the candidate point being used in the Least Squares
54  // approximation (X_i-X_0).
55 
56  Matrix A(nvals-1, D);
57  for (int i = 0; i < nvals-1; ++i) {
58  for (int j = 0; j < D; ++j)
59  A[i][j] = coords[i+1][j]-coord0[j];
60  }
61 
62 
63  // A is a matrix of size nvals-1 by D (where D is the space
64  // dimension). So transpose(A)*A is D by D
65 
66  Matrix AT = A.transpose();
67 
68  Matrix ATA = AT*A;
69 
70  // Each entry/row of F contains the difference between the
71  // function value at the candidate point and the function value
72  // at the point where we are computing (f-f_0)
73 
74  std::vector<double> F(nvals-1);
75  for (int i = 0; i < nvals-1; ++i)
76  F[i] = vals[i+1]-val0;
77

79  // (where D is the space dimension)
